507  void
508  message_queue::internal_construct_ (std::size_t msgs,
509  std::size_t msg_size_bytes,
510  const attributes& attr,
511  void* queue_address,
512  std::size_t queue_size_bytes)
513  {
514  // Don't call this from interrupt handlers.
515  os_assert_throw(!interrupts::in_handler_mode (), EPERM);
516 
517 #if !defined(OS_USE_RTOS_PORT_MESSAGE_QUEUE)
518  clock_ = attr.clock != nullptr ? attr.clock : &sysclock;
519 #endif
520  msg_size_bytes_ = static_cast<message_queue::msg_size_t> (msg_size_bytes);
521  assert(msg_size_bytes_ == msg_size_bytes);
522  assert(msg_size_bytes_ > 0);
523 
524  // in order for the list of free messages to not consume additional memory,
525  // the pointers are stored at the beginning of the messages, thus messages should be large enough to fit a pointer
526  assert(msg_size_bytes_ >= sizeof(void*));
527 
528  msgs_ = static_cast<message_queue::size_t> (msgs);
529  assert(msgs_ == msgs);
530  assert(msgs > 0);
531 
532  // If the storage is given explicitly, override attributes.
533  if (queue_address != nullptr)
534  {
535  // The attributes should not define any storage in this case.
536  assert(attr.mq_queue_address == nullptr);
537 
538  queue_addr_ = queue_address;
539  queue_size_bytes_ = queue_size_bytes;
540  }
541  else
542  {
543  queue_addr_ = attr.mq_queue_address;
544  queue_size_bytes_ = attr.mq_queue_size_bytes;
545  }
546 
547 #if defined(OS_TRACE_RTOS_MQUEUE)
548  trace::printf ("%s() @%p %s %u %u %p %u\n", __func__, this, name (),
549  msgs_, msg_size_bytes_, queue_addr_, queue_size_bytes_);
550 #endif
551 
552 #if !defined(OS_USE_RTOS_PORT_MESSAGE_QUEUE)
553  std::size_t storage_size = compute_allocated_size_bytes<void*> (
554  msgs, msg_size_bytes);
555 #endif
556  if (queue_addr_ != nullptr)
557  {
558  // The queue must be real, and have a non zero size.
559  os_assert_throw(queue_size_bytes_ > 0, EINVAL);
560 #if defined(OS_USE_RTOS_PORT_MESSAGE_QUEUE)
561  os_assert_throw(
562  queue_size_bytes_ >= (std::size_t) (msgs * msg_size_bytes),
563  EINVAL);
564 #else
565  // The queue must fit the storage.
566  os_assert_throw(queue_size_bytes_ >= storage_size, EINVAL);
567 #endif
568  }
569 
570 #if defined(OS_USE_RTOS_PORT_MESSAGE_QUEUE)
571 
572  count_ = 0;
573  port::message_queue::create (this);
574 
575 #else
576 
577  head_ = no_index;
578 
579  // The queue storage must have a real address.
580  os_assert_throw(queue_addr_ != nullptr, ENOMEM);
581 
582  // The array of prev indexes follows immediately after the content array.
583  prev_array_ =
584  reinterpret_cast<index_t*> (static_cast<char*> (queue_addr_)
585  + msgs
586  * ((msg_size_bytes + (sizeof(void*) - 1))
587  & ~(sizeof(void*) - 1)));
588  // The array of next indexes follows immediately the prev array.
589  next_array_ =
590  reinterpret_cast<index_t*> (reinterpret_cast<char*> (const_cast<index_t*> (prev_array_))
591  + msgs * sizeof(index_t));
592  // The array of priorities follows immediately the next array.
593  prio_array_ =
594  reinterpret_cast<priority_t*> (reinterpret_cast<char*> (const_cast<index_t*> (next_array_))
595  + msgs * sizeof(index_t));
596 
597 #if !defined(NDEBUG)
598  char* p =
599  reinterpret_cast<char*> (reinterpret_cast<char*> (const_cast<priority_t*> (prio_array_))
600  + msgs * sizeof(priority_t));
601 
602  assert(
603  p - static_cast<char*> (queue_addr_)
604  <= static_cast<ptrdiff_t> (queue_size_bytes_));
605 #endif
606 
607  internal_init_ ();
608 #endif

611  void
612  message_queue::internal_init_ (void)
613  {
614  count_ = 0;
615 
616 #if !defined(OS_USE_RTOS_PORT_MESSAGE_QUEUE)
617 
618  // Construct a linked list of blocks. Store the pointer at
619  // the beginning of each block. Each block
620  // will hold the address of the next free block,
621  // or `nullptr` at the end.
622  char* p = static_cast<char*> (queue_addr_);
623  for (std::size_t i = 1; i < msgs_; ++i)
624  {
625  // Compute the address of the next block;
626  char* pn = p + msg_size_bytes_;
627 
628  // Make this block point to the next one.
629  *(static_cast<void**> (static_cast<void*> (p))) = pn;
630  // Advance pointer
631  p = pn;
632  }
633 
634  // Mark end of list.
635  *(static_cast<void**> (static_cast<void*> (p))) = nullptr;
636 
637  first_free_ = queue_addr_; // Pointer to first block.
638 
639  head_ = no_index;
640 
641  // Need not be inside the critical section,
642  // the lists are protected by inner `resume_one()`.
643 
644  // Wake-up all threads, if any.
645  send_list_.resume_all ();
646  receive_list_.resume_all ();

655  * Internal function.
656  * Should be called from an interrupts critical section.
657  */
658  bool
659  message_queue::internal_try_send_ (const void* msg, std::size_t nbytes,
660  priority_t mprio)
661  {
662  if (first_free_ == nullptr)
663  {
664  // No available space to send the message.
665  return false;
666  }
667 
668  // The first step is to remove the free block from the list,
669  // so another concurrent call will not get it too.
670 
671  // Get the address where the message will be copied.
672  // This is the first free memory block.
673  char* dest = static_cast<char*> (first_free_);
674 
675  // Update to next free, if any (the last one has nullptr).
676  first_free_ = *(static_cast<void**> (first_free_));
677 
678  // The second step is to copy the message from the user buffer.
679  {
680  // ----- Enter uncritical section -----------------------------------
681  // interrupts::uncritical_section iucs;
682 
683  // Copy message from user buffer to queue storage.
684  std::memcpy (dest, msg, nbytes);
685  if (nbytes < msg_size_bytes_)
686  {
687  // Fill in the remaining space with 0x00.
688  std::memset (dest + nbytes, 0x00, msg_size_bytes_ - nbytes);
689  }
690  // ----- Exit uncritical section ------------------------------------
691  }
692 
693  // The third step is to link the buffer to the list.
694 
695  // Using the address, compute the index in the array.
696  std::size_t msg_ix = (static_cast<std::size_t> (dest
697  - static_cast<char*> (queue_addr_)) / msg_size_bytes_);
698  prio_array_[msg_ix] = mprio;
699 
700  if (head_ == no_index)
701  {
702  // No other message in the queue, enlist this one
703  // as head, with links to itself.
704  head_ = static_cast<index_t> (msg_ix);
705  prev_array_[msg_ix] = static_cast<index_t> (msg_ix);
706  next_array_[msg_ix] = static_cast<index_t> (msg_ix);
707  }
708  else
709  {
710  std::size_t ix;
711  // Arrange to insert between head and tail.
712  ix = prev_array_[head_];
713  // Check if the priority is higher than the head priority.
714  if (mprio > prio_array_[head_])
715  {
716  // Having the highest priority, the new message
717  // becomes the new head.
718  head_ = static_cast<index_t> (msg_ix);
719  }
720  else
721  {
722  // If not higher than the head, try to insert at the tail,
723  // but advance up until the same priority is found.
724  while ((mprio > prio_array_[ix]))
725  {
726  ix = prev_array_[ix];
727  }
728  }
729  prev_array_[msg_ix] = static_cast<index_t> (ix);
730  next_array_[msg_ix] = next_array_[ix];
731 
732  // Break the chain and insert the new index.
733  std::size_t tmp_ix = next_array_[ix];
734  next_array_[ix] = static_cast<index_t> (msg_ix);
735  prev_array_[tmp_ix] = static_cast<index_t> (msg_ix);
736  }
737 
738  // One more message added to the queue.
739  ++count_;
740 
741  // Wake-up one thread, if any.
742  receive_list_.resume_one ();
743 
744  return true;
745  }

751  /*
752  * Internal function.
753  * Should be called from an interrupts critical section.
754  */
755  bool
756  message_queue::internal_try_receive_ (void* msg, std::size_t nbytes,
757  priority_t* mprio)
758  {
759 
760  if (head_ == no_index)
761  {
762  return false;
763  }
764 
765  // Compute the message source address.
766  char* src = static_cast<char*> (queue_addr_) + head_ * msg_size_bytes_;
767  priority_t prio = prio_array_[head_];
768 
769 #if defined(OS_TRACE_RTOS_MQUEUE_)
770  trace::printf ("%s(%p,%u) @%p %s src %p %p\n", __func__, msg, nbytes,
771  this, name (), src, first_free_);
772 #endif
773 
774  // Unlink it from the list, so another concurrent call will
775  // not get it too.
776  if (count_ > 1)
777  {
778  // Remove the current element from the list.
779  prev_array_[next_array_[head_]] = prev_array_[head_];
780  next_array_[prev_array_[head_]] = next_array_[head_];
781 
782  // Next becomes the new head.
783  head_ = next_array_[head_];
784  }
785  else
786  {
787  // If there was only one, the list is empty now.
788  head_ = no_index;
789  }
790 
791  --count_;
792 
793  // Copy to destination
794  {
795  // ----- Enter uncritical section -----------------------------------
796  interrupts::uncritical_section iucs;
797 
798  // Copy message from queue to user buffer.
799  memcpy (msg, src, nbytes);
800  if (mprio != nullptr)
801  {
802  *mprio = prio;
803  }
804  // ----- Exit uncritical section ------------------------------------
805  }
806 
807  // After the message was copied, the block can be released.
808 
809  // Perform a push_front() on the single linked LIFO list,
810  // i.e. add the block to the beginning of the list.
811 
812  // Link previous list to this block; may be null, but it does
813  // not matter.
814  *(static_cast<void**> (static_cast<void*> (src))) = first_free_;
815 
816  // Now this block is the first one.
817  first_free_ = src;
818 
819  // Wake-up one thread, if any.
820  send_list_.resume_one ();
821 
822  return true;
823  }

869  result_t
870  message_queue::send (const void* msg, std::size_t nbytes, priority_t mprio)
871  {
872 #if defined(OS_TRACE_RTOS_MQUEUE)
873  trace::printf ("%s(%p,%d,%d) @%p %s\n", __func__, msg, nbytes, mprio,
874  this, name ());
875 #endif
876 
877  // Don't call this from interrupt handlers.
878  os_assert_err(!interrupts::in_handler_mode (), EPERM);
879  // Don't call this from critical regions.
880  os_assert_err(!scheduler::locked (), EPERM);
881 
882  os_assert_err(msg != nullptr, EINVAL);
883  os_assert_err(nbytes <= msg_size_bytes_, EMSGSIZE);
884 
885 #if defined(OS_USE_RTOS_PORT_MESSAGE_QUEUE)
886 
887  return port::message_queue::send (this, msg, nbytes, mprio);
888 
889 #else
890 
891  {
892  // ----- Enter critical section -------------------------------------
893  interrupts::critical_section ics;
894 
895  if (internal_try_send_ (msg, nbytes, mprio))
896  {
897  return result::ok;
898  }
899  // ----- Exit critical section --------------------------------------
900  }
901 
902  thread& crt_thread = this_thread::thread ();
903 
904  // Prepare a list node pointing to the current thread.
905  // Do not worry for being on stack, it is temporarily linked to the
906  // list and guaranteed to be removed before this function returns.
907  internal::waiting_thread_node node
908  { crt_thread };
909 
910  for (;;)
911  {
912  {
913  // ----- Enter critical section ---------------------------------
914  interrupts::critical_section ics;
915 
916  if (internal_try_send_ (msg, nbytes, mprio))
917  {
918  return result::ok;
919  }
920 
921  // Add this thread to the message queue send waiting list.
922  scheduler::internal_link_node (send_list_, node);
923  // state::suspended set in above link().
924  // ----- Exit critical section ----------------------------------
925  }
926 
927  port::scheduler::reschedule ();
928 
929  // Remove the thread from the message queue send waiting list,
930  // if not already removed by receive().
931  scheduler::internal_unlink_node (node);
932 
933  if (crt_thread.interrupted ())
934  {
935 #if defined(OS_TRACE_RTOS_MQUEUE)
936  trace::printf ("%s(%p,%d,%d) EINTR @%p %s\n", __func__, msg,
937  nbytes, mprio, this, name ());
938 #endif
939  return EINTR;
940  }
941  }
942 
943  /* NOTREACHED */
944  return ENOTRECOVERABLE;
945 
946 #endif
947  }

1124  return result::ok;
1125  }
1126

1156 #endif

1158  }
