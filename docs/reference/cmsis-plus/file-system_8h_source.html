818  template<typename T>
819  T*
820  file_system::allocate_file (void)
821  {
822  using file_type = T;
823 
824  file_type* fil;
825 
826  if (deferred_files_list_.empty ())
827  {
828  fil = new file_type (*this);
829  }
830  else
831  {
832  fil = static_cast<file_type*> (deferred_files_list_.unlink_head ());
833 
834  // Call the constructor before reusing the object,
835  fil->~file_type ();
836 
837  // Placement new, run only the constructor.
838  new (fil) file_type (*this);
839 
840  deallocate_files<file_type> ();
841  }
842  return fil;
843  }
