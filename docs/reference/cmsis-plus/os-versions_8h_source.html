33// According to Semantic Versioning 2.0.0 (http://semver.org),
34// given a version number MAJOR.MINOR.PATCH, increment the:
35//
36// MAJOR version when you make incompatible API changes,
37// MINOR version when you add functionality in a backwards-compatible manner,
38// PATCH version when you make backwards-compatible bug fixes,
39// PRE_RELEASE is a hyphen and a series of dot separated identifiers,
40// BUILD a plus sign and a series of dot separated identifiers.
41
42// ----------------------------------------------------------------------------
43
44#define OS_MACRO_SHARP(x) #x
45#define OS_MACRO_STRINGIFY(x) OS_MACRO_SHARP(x)
