511
512 // Don't call this from interrupt handlers.
513 os_assert_err(!interrupts::in_handler_mode (), EPERM);
514 // Don't call this from critical regions.
515 os_assert_err(!scheduler::locked (), EPERM);
516
517 thread& crt_thread = this_thread::thread ();
518
519 // Prepare a list node pointing to the current thread.
520 // Do not worry for being on stack, it is temporarily linked to the
521 // list and guaranteed to be removed before this function returns.
522 internal::waiting_thread_node node
523 { crt_thread };
524

527 result_t res;
528 res = mutex.unlock ();
529
530 if (res != result::ok)
531 {
532 return res;
533 }
534
535 {
536 // Add this thread to the condition variable waiting list.
537 list_.link (node);
538 node.thread_->waiting_node_ = &node;

543 // if not already removed.
