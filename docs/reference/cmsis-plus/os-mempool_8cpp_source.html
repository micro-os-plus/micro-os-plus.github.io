308 void
309 memory_pool::internal_construct_ (std::size_t blocks,
310 std::size_t block_size_bytes,
311 const attributes& attr,
312 void* pool_address,
313 std::size_t pool_size_bytes)
314 {
315 // Don't call this from interrupt handlers.
316 os_assert_throw(!interrupts::in_handler_mode (), EPERM);
317
318#if !defined(OS_USE_RTOS_PORT_MEMORY_POOL)
319 clock_ = attr.clock != nullptr ? attr.clock : &sysclock;
320#endif
321
322 blocks_ = static_cast<memory_pool::size_t> (blocks);
323 assert(blocks_ == blocks);
324 assert(blocks_ > 0);
325
326 // Adjust block size to multiple of pointer.
327 // Blocks must be large enough to store a pointer, used
328 // to construct the list of free blocks.
329 block_size_bytes_ = (static_cast<memory_pool::size_t> (block_size_bytes
330 + __SIZEOF_POINTER__ - 1))
331 & (static_cast<memory_pool::size_t> (~(__SIZEOF_POINTER__ - 1)));
332
333 // If the storage is given explicitly, override attributes.
334 if (pool_address != nullptr)
335 {
336 // The attributes should not define any storage in this case.
337 assert(attr.mp_pool_address == nullptr);
338
339 pool_addr_ = pool_address;
340 pool_size_bytes_ = pool_size_bytes;
341 }
342 else
343 {
344 pool_addr_ = attr.mp_pool_address;
345 pool_size_bytes_ = attr.mp_pool_size_bytes;
346 }
347
348 // Blocks must be pointer aligned.
349 void* p = pool_addr_;
350 std::size_t sz = pool_size_bytes_;
351 pool_addr_ = static_cast<char*> (std::align (__SIZEOF_POINTER__,
352 blocks_ * block_size_bytes_,
353 p, sz));
354
355#if defined(OS_TRACE_RTOS_MEMPOOL)
356 trace::printf ("%s() @%p %s %u %u %p %u\n", __func__, this, name (),
357 blocks_, block_size_bytes_, pool_addr_, pool_size_bytes_);
358#endif
359
360 std::size_t storage_size = compute_allocated_size_bytes<void*> (
361 blocks_, block_size_bytes_);
362
363 if (pool_addr_ != nullptr)
364 {
365 // The pool must be real, and have a non zero size.
366 os_assert_throw(pool_size_bytes_ > 0, EINVAL);
367 // The pool must fit the storage.
368 os_assert_throw(pool_size_bytes_ >= storage_size, EINVAL);
369 }
370
371 // The pool must have a real address.
372 os_assert_throw(pool_addr_ != nullptr, ENOMEM);
373
374 internal_init_ ();
375
376 }

421 /*
422 * Construct the linked list of blocks and initialise the
423 * internal pointers and counters.
424 */
425 void
426 memory_pool::internal_init_ (void)
427 {
428 // Construct a linked list of blocks. Store the pointer at
429 // the beginning of each block. Each block
430 // will hold the address of the next free block, or nullptr at the end.
431 char* p = static_cast<char*> (pool_addr_);
432 for (std::size_t i = 1; i < blocks_; ++i)
433 {
434 // Compute the address of the next block;
435 char* pn = p + block_size_bytes_;
436
437 // Make this block point to the next one.
438 *(static_cast<void**> (static_cast<void*> (p))) = pn;
439 // Advance pointer
440 p = pn;
441 }
442
443 // Mark end of list.
444 *(static_cast<void**> (static_cast<void*> (p))) = nullptr;
445
446 first_ = pool_addr_; // Pointer to first block.
447
448 count_ = 0; // No allocated blocks.
449 }

451 /*
452 * Internal function used to return the first block in the
453 * free list.
454 * Should be called from an interrupts critical section.
455 */
456 void*
457 memory_pool::internal_try_first_ (void)
458 {
459 if (first_ != nullptr)
460 {
461 void* p = static_cast<void*> (first_);
462 first_ = *(static_cast<void**> (first_));
463 ++count_;
464 return p;
465 }
466
467 return nullptr;
468 }

493 void*
494 memory_pool::alloc (void)
495 {
496#if defined(OS_TRACE_RTOS_MEMPOOL)
497 trace::printf ("%s() @%p %s\n", __func__, this, name ());
498#endif
499
500 // Don't call this from interrupt handlers.
501 os_assert_throw(!interrupts::in_handler_mode (), EPERM);
502 // Don't call this from critical regions.
503 os_assert_throw(!scheduler::locked (), EPERM);
504
505 void* p;
506
507 // Extra test before entering the loop, with its inherent weight.
508 // Trade size for speed.
509 {
510 // ----- Enter critical section -------------------------------------
511 interrupts::critical_section ics;
512
513 p = internal_try_first_ ();
514 if (p != nullptr)
515 {
516#if defined(OS_TRACE_RTOS_MEMPOOL)
517 trace::printf ("%s()=%p @%p %s\n", __func__, p, this, name ());
518#endif
519 return p;
520 }
521 // ----- Exit critical section --------------------------------------
522 }

524 thread& crt_thread = this_thread::thread ();
525
526 // Prepare a list node pointing to the current thread.
527 // Do not worry for being on stack, it is temporarily linked to the
528 // list and guaranteed to be removed before this function returns.
529 internal::waiting_thread_node node
530 { crt_thread };
531
532 for (;;)
533 {
534 {
535 // ----- Enter critical section ---------------------------------
536 interrupts::critical_section ics;
537
538 p = internal_try_first_ ();
539 if (p != nullptr)
540 {
541#if defined(OS_TRACE_RTOS_MEMPOOL)
542 trace::printf ("%s()=%p @%p %s\n", __func__, p, this,
543 name ());
544#endif
545 return p;
546 }
547
548 // Add this thread to the memory pool waiting list.
549 scheduler::internal_link_node (list_, node);
550 // state::suspended set in above link().
551 // ----- Exit critical section ----------------------------------
552 }
553
554 port::scheduler::reschedule ();
555
556 // Remove the thread from the memory pool waiting list,
557 // if not already removed by free().
558 scheduler::internal_unlink_node (node);
559
560 if (this_thread::thread ().interrupted ())
561 {
562#if defined(OS_TRACE_RTOS_MEMPOOL)
563 trace::printf ("%s() INTR @%p %s\n", __func__, this, name ());
564#endif

655 void*
656 memory_pool::timed_alloc (clock::duration_t timeout)
657 {
658#if defined(OS_TRACE_RTOS_MEMPOOL)
659 trace::printf ("%s(%u) @%p %s\n", __func__,
660 static_cast<unsigned int> (timeout), this, name ());
661#endif
662
663 // Don't call this from interrupt handlers.
664 os_assert_throw(!interrupts::in_handler_mode (), EPERM);
665 // Don't call this from critical regions.
666 os_assert_throw(!scheduler::locked (), EPERM);
667
668 void* p;
669
670 // Extra test before entering the loop, with its inherent weight.
671 // Trade size for speed.
672 {
673 // ----- Enter critical section -------------------------------------
674 interrupts::critical_section ics;
675
676 p = internal_try_first_ ();
677 if (p != nullptr)
678 {
679#if defined(OS_TRACE_RTOS_MEMPOOL)
680 trace::printf ("%s()=%p @%p %s\n", __func__, p, this, name ());
681#endif
682 return p;
683 }
684 // ----- Exit critical section --------------------------------------
685 }
686
687 thread& crt_thread = this_thread::thread ();
688
689 // Prepare a list node pointing to the current thread.
690 // Do not worry for being on stack, it is temporarily linked to the
691 // list and guaranteed to be removed before this function returns.
692 internal::waiting_thread_node node
693 { crt_thread };
694
695 internal::clock_timestamps_list& clock_list = clock_->steady_list ();
696 clock::timestamp_t timeout_timestamp = clock_->steady_now () + timeout;
697
698 // Prepare a timeout node pointing to the current thread.
699 internal::timeout_thread_node timeout_node
700 { timeout_timestamp, crt_thread };
701
702 for (;;)
703 {
704 {
705 // ----- Enter critical section ---------------------------------
706 interrupts::critical_section ics;
707
708 p = internal_try_first_ ();
709 if (p != nullptr)
710 {
711#if defined(OS_TRACE_RTOS_MEMPOOL)
712 trace::printf ("%s()=%p @%p %s\n", __func__, p, this,
713 name ());
714#endif
715 return p;
716 }
717
718 // Add this thread to the memory pool waiting list,
719 // and the clock timeout list.
720 scheduler::internal_link_node (list_, node, clock_list,
721 timeout_node);
722 // state::suspended set in above link().
723 // ----- Exit critical section ----------------------------------
724 }
725
726 port::scheduler::reschedule ();
727
728 // Remove the thread from the memory pool waiting list,
729 // if not already removed by free() and from the clock
730 // timeout list, if not already removed by the timer.
731 scheduler::internal_unlink_node (node, timeout_node);
732
733 if (this_thread::thread ().interrupted ())
734 {
735#if defined(OS_TRACE_RTOS_MEMPOOL)
736 trace::printf ("%s() INTR @%p %s\n", __func__, this, name ());
737#endif
738 return nullptr;
739 }
740
741 if (clock_->steady_now () >= timeout_timestamp)
742 {
743#if defined(OS_TRACE_RTOS_MEMPOOL)
744 trace::printf ("%s() TMO @%p %s\n", __func__, this, name ());
745#endif
746 return nullptr;
747 }
748 }
749
750 /* NOTREACHED */

763 result_t
764 memory_pool::free (void* block)
765 {
766#if defined(OS_TRACE_RTOS_MEMPOOL)
767 trace::printf ("%s(%p) @%p %s\n", __func__, block, this, name ());
768#endif
769
770 // Don't call this from high priority interrupts.
771 assert(port::interrupts::is_priority_valid ());
772
773 // Validate pointer.
774 if ((block < pool_addr_)
775 || (block
776 >= (static_cast<char*> (pool_addr_) + blocks_ * block_size_bytes_)))
777 {
778#if defined(OS_TRACE_RTOS_MEMPOOL)
779 trace::printf ("%s(%p) EINVAL @%p %s\n", __func__, block, this,
780 name ());
781#endif
782 return EINVAL;
783 }
784
785 {
786 // ----- Enter critical section -------------------------------------
787 interrupts::critical_section ics;
788
789 // Perform a push_front() on the single linked LIFO list,
790 // i.e. add the block to the beginning of the list.
791
792 // Link previous list to this block; may be null, but it does
793 // not matter.
794 *(static_cast<void**> (block)) = first_;
795
796 // Now this block is the first one.
797 first_ = block;
798
799 --count_;
800 // ----- Exit critical section --------------------------------------
801 }
802
803 // Wake-up one thread, if any.
804 list_.resume_one ();
805
806 return result::ok;
807 }

816 memory_pool::reset (void)
817 {
818#if defined(OS_TRACE_RTOS_MEMPOOL)
819 trace::printf ("%s() @%p %s\n", __func__, this, name ());
820#endif
821
822 // Don't call this from interrupt handlers.
823 os_assert_err(!interrupts::in_handler_mode (), EPERM);
824
825 {
826 // ----- Enter critical section -------------------------------------
827 interrupts::critical_section ics;
828
829 internal_init_ ();
830 // ----- Exit critical section --------------------------------------
831 }
832
833 // Wake-up all threads, if any.
834 // Need not be inside the critical section,
835 // the list is protected by inner `resume_one()`.
836 list_.resume_all ();
837
838 return result::ok;
