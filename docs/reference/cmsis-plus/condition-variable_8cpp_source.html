64 void
65 condition_variable::wait (std::unique_lock<mutex>& lk)
66 {
67 if (!lk.owns_lock ())
68 {
69 os::estd::__throw_system_error (
70 EPERM, "condition_variable::wait: mutex not locked");
71 }
72 os::rtos::result_t res = ncv_.wait ((*(lk.mutex ()->native_handle ())));
73 if (res != os::rtos::result::ok)
74 {
75 os::estd::__throw_cmsis_error (static_cast<int> (res),
76 "condition_variable wait failed");
77 }
78 }
