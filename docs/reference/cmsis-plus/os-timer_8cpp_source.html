246 timer::start (clock::duration_t period)
247 {
248#if defined(OS_TRACE_RTOS_TIMER)
249 trace::printf ("%s(%u) @%p %s\n", __func__,
250 static_cast<unsigned int> (period), this, name ());
251#endif
252
253 // Don't call this from interrupt handlers.
254 os_assert_err(!interrupts::in_handler_mode (), EPERM);
255
256 if (period == 0)
257 {
258 period = 1;
259 }
260
261 result_t res;
262
263#if defined(OS_USE_RTOS_PORT_TIMER)
264
265 res = port::timer::start (this, period);
266
267#else
268
269 period_ = period;
270
271 timer_node_.timestamp = clock_->steady_now () + period;
272
273 {
274 // ----- Enter critical section -------------------------------------
275 interrupts::critical_section ics;
276
277 // If started, stop.
278 timer_node_.unlink ();
279
280 clock_->steady_list ().link (timer_node_);
281 // ----- Exit critical section --------------------------------------
282 }
283 res = result::ok;
284
285#endif
286
287 if (res == result::ok)
288 {
289 state_ = state::running;
290 }
291 return res;
