83void*
84malloc (size_t bytes)
85{
86 assert(!rtos::interrupts::in_handler_mode ());
87
88 void* mem;
89 {
90 // ----- Begin of critical section --------------------------------------
91 rtos::scheduler::critical_section scs;
92
93 errno = 0;
94 mem = estd::pmr::get_default_resource ()->allocate (bytes);
95 if (mem == nullptr)
96 {
97 errno = ENOMEM;
98 }
99
100#if defined(OS_TRACE_LIBC_MALLOC)
101 trace::printf ("::%s(%d)=%p\n", __func__, bytes, mem);
102#endif
103 // ----- End of critical section ----------------------------------------
104 }
105
106 return mem;
107}

150void*
151calloc (size_t nelem, size_t elbytes)
152{
153 assert(!rtos::interrupts::in_handler_mode ());
154
155 errno = 0;
156 if (nelem == 0 || elbytes == 0)
157 {
158 return nullptr;
159 }
160
161 void* mem;
162 {
163 // ----- Begin of critical section --------------------------------------
164 rtos::scheduler::critical_section scs;
165
166 mem = estd::pmr::get_default_resource ()->allocate (nelem * elbytes);
167
168#if defined(OS_TRACE_LIBC_MALLOC)
169 trace::printf ("::%s(%u,%u)=%p\n", __func__, nelem, elbytes, mem);
170#endif
171 // ----- End of critical section ----------------------------------------
172 }
173
174 if (mem != nullptr)
175 {
176 memset (mem, 0, nelem * elbytes);
177 }
178 else
179 {
180 errno = ENOMEM;
181 }
182
183 return mem;
184}

340void
341free (void* ptr)
342{
343 assert(!rtos::interrupts::in_handler_mode ());
344
345 if (ptr == nullptr)
346 {
347 return;
348 }
349
350 // ----- Begin of critical section ------------------------------------------
351 rtos::scheduler::critical_section scs;
352
353#if defined(OS_TRACE_LIBC_MALLOC)
354 trace::printf ("::%s(%p)\n", __func__, ptr);
355#endif
356
357 // Size unknown, pass 0.
358 estd::pmr::get_default_resource ()->deallocate (ptr, 0);
359 // ----- End of critical section --------------------------------------------
360}
