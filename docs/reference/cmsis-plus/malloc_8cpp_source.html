94void*
95malloc (size_t bytes)
96{
97 assert(!rtos::interrupts::in_handler_mode ());
98
99 void* mem;
100 {
101 // ----- Begin of critical section --------------------------------------
102 rtos::scheduler::critical_section scs;
103
104 errno = 0;
105 mem = estd::pmr::get_default_resource ()->allocate (bytes);
106 if (mem == nullptr)
107 {
108 errno = ENOMEM;
109 }
110
111#if defined(OS_TRACE_LIBC_MALLOC)
112 trace::printf ("::%s(%d)=%p\n", __func__, bytes, mem);
113#endif
114 // ----- End of critical section ----------------------------------------
115 }
116
117 return mem;
118}

161void*
162calloc (size_t nelem, size_t elbytes)
163{
164 assert(!rtos::interrupts::in_handler_mode ());
165
166 errno = 0;
167 if (nelem == 0 || elbytes == 0)
168 {
169 return nullptr;
170 }
171
172 void* mem;
173 {
174 // ----- Begin of critical section --------------------------------------
175 rtos::scheduler::critical_section scs;
176
177 mem = estd::pmr::get_default_resource ()->allocate (nelem * elbytes);
178
179#if defined(OS_TRACE_LIBC_MALLOC)
180 trace::printf ("::%s(%u,%u)=%p\n", __func__, nelem, elbytes, mem);
181#endif
182 // ----- End of critical section ----------------------------------------
183 }
184
185 if (mem != nullptr)
186 {
187 memset (mem, 0, nelem * elbytes);
188 }
189 else
190 {
191 errno = ENOMEM;
192 }
193
194 return mem;
195}

260void*
261realloc (void* ptr, size_t bytes)
262{
263 assert(!rtos::interrupts::in_handler_mode ());
264
265 void* mem;
266
267 {
268 // ----- Begin of critical section --------------------------------------
269 rtos::scheduler::critical_section scs;
270
271 errno = 0;
272 if (ptr == nullptr)
273 {
274 mem = estd::pmr::get_default_resource ()->allocate (bytes);
275#if defined(OS_TRACE_LIBC_MALLOC)
276 trace::printf ("::%s(%p,%u)=%p\n", __func__, ptr, bytes, mem);
277#endif
278 if (mem == nullptr)
279 {
280 errno = ENOMEM;
281 }
282 return mem;
283 }
284
285 if (bytes == 0)
286 {
287 estd::pmr::get_default_resource ()->deallocate (ptr, 0);
288#if defined(OS_TRACE_LIBC_MALLOC)
289 trace::printf ("::%s(%p,%u)=0\n", __func__, ptr, bytes);
290#endif
291 return nullptr;
292 }
293
294#if 0
295 /* TODO: There is chance to shrink the chunk if newly requested
296 * size is much small */
297 if (nano_malloc_usable_size (RCALL ptr) >= bytes)
298 return ptr;
299#endif
300
301 mem = estd::pmr::get_default_resource ()->allocate (bytes);
302 if (mem != nullptr)
303 {
304 memcpy (mem, ptr, bytes);
305 estd::pmr::get_default_resource ()->deallocate (ptr, 0);
306 }
307 else
308 {
309 errno = ENOMEM;

313 trace::printf ("::%s(%p,%u)=%p", __func__, ptr, bytes, mem);

351void
352free (void* ptr)
353{
354 assert(!rtos::interrupts::in_handler_mode ());
355
356 if (ptr == nullptr)
357 {
358 return;
359 }
360
361 // ----- Begin of critical section ------------------------------------------
362 rtos::scheduler::critical_section scs;
363
364#if defined(OS_TRACE_LIBC_MALLOC)
365 trace::printf ("::%s(%p)\n", __func__, ptr);
366#endif
367
368 // Size unknown, pass 0.
369 estd::pmr::get_default_resource ()->deallocate (ptr, 0);
370 // ----- End of critical section --------------------------------------------
371}
