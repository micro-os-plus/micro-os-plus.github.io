239 #if defined(OS_TRACE_RTOS_EVFLAGS)

245  os_assert_throw(!interrupts::in_handler_mode (), EPERM);

247  os_assert_throw(!scheduler::locked (), EPERM);
248

258

260  {
261 #if defined(OS_TRACE_RTOS_EVFLAGS)
262  trace::printf ("%s(0x%X,%u) @%p %s >0x%X\n", __func__, mask, mode,

266  }

268  }

273  // Do not worry for being on stack, it is temporarily linked to the
274  // list and guaranteed to be removed before this function returns.
275  internal::waiting_thread_node node
276  { crt_thread };
277

420  result_t
421  event_flags::timed_wait (flags::mask_t mask, clock::duration_t timeout,
422  flags::mask_t* oflags, flags::mode_t mode)
423  {
424 #if defined(OS_TRACE_RTOS_EVFLAGS)
425  trace::printf ("%s(0x%X,%u,%u) @%p %s <0x%X\n", __func__, mask, timeout,
426  mode, this, name (), event_flags_.mask ());
427 #endif
428 
429  // Don't call this from interrupt handlers.
430  os_assert_throw(!interrupts::in_handler_mode (), EPERM);
431  // Don't call this from critical regions.
432  os_assert_throw(!scheduler::locked (), EPERM);
433 
434 #if defined(OS_USE_RTOS_PORT_EVENT_FLAGS)
435 
436  return port::event_flags::timed_wait (this, mask, timeout, oflags, mode);
437 
438 #else
439 
440  // Extra test before entering the loop, with its inherent weight.
441  // Trade size for speed.
442  {
443  // ----- Enter critical section -------------------------------------
444  interrupts::critical_section ics;
445 
446  if (event_flags_.check_raised (mask, oflags, mode))
447  {
448 #if defined(OS_TRACE_RTOS_EVFLAGS)
449  trace::printf ("%s(0x%X,%u,%u) @%p %s >0x%X\n", __func__, mask,
450  timeout, mode, this, name (),
451  event_flags_.mask ());
452 #endif
453  return result::ok;
454  }
455  // ----- Exit critical section --------------------------------------
456  }
457 
458  thread& crt_thread = this_thread::thread ();
459 
460  // Prepare a list node pointing to the current thread.
461  // Do not worry for being on stack, it is temporarily linked to the
462  // list and guaranteed to be removed before this function returns.
463  internal::waiting_thread_node node
464  { crt_thread };
465 
466  internal::clock_timestamps_list& clock_list = clock_->steady_list ();
467  clock::timestamp_t timeout_timestamp = clock_->steady_now () + timeout;
468 
469  // Prepare a timeout node pointing to the current thread.
470  internal::timeout_thread_node timeout_node
471  { timeout_timestamp, crt_thread };
472 
473  for (;;)
474  {
475  {
476  // ----- Enter critical section ---------------------------------
477  interrupts::critical_section ics;
478 
479  if (event_flags_.check_raised (mask, oflags, mode))
480  {
481 #if defined(OS_TRACE_RTOS_EVFLAGS)
482  trace::printf ("%s(0x%X,%u,%u) @%p %s >0x%X\n", __func__,
483  mask, timeout, mode, this, name (),
484  event_flags_.mask ());
485 #endif
486  return result::ok;
487  }
488 
489  // Add this thread to the event flags waiting list,
490  // and the clock timeout list.
491  scheduler::internal_link_node (list_, node, clock_list,
492  timeout_node);
493  // state::suspended set in above link().
494  // ----- Exit critical section ----------------------------------
495  }
496 
497  port::scheduler::reschedule ();
498 
499  // Remove the thread from the event flags waiting list,
500  // if not already removed by raise() and from the clock
501  // timeout list, if not already removed by the timer.
502  scheduler::internal_unlink_node (node, timeout_node);
503 
504  if (crt_thread.interrupted ())
505  {
506 #if defined(OS_TRACE_RTOS_EVFLAGS)
507  trace::printf ("%s(0x%X,%u,%u) EINTR @%p %s 0x%X \n", __func__,
508  mask, timeout, mode, this, name ());
509 #endif
510  return EINTR;
511  }
512 
513  if (clock_->steady_now () >= timeout_timestamp)
514  {
515 #if defined(OS_TRACE_RTOS_EVFLAGS)
516  trace::printf ("%s(0x%X,%u,%u) ETIMEDOUT @%p %s 0x%X \n",
517  __func__, mask, timeout, mode, this, name ());
518 #endif
519  return ETIMEDOUT;
520  }
521  }
522 
523  /* NOTREACHED */
524  return ENOTRECOVERABLE;
525 
526 #endif
527  }
