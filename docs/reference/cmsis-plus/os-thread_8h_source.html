2523 template<typename Allocator>
2524 thread_allocated<Allocator>::thread_allocated (
2525 const char* name, func_t function, func_args_t args,
2526 const attributes& attr, const allocator_type& allocator) :
2527 thread
2528 { name }
2529 {
2530#if defined(OS_TRACE_RTOS_THREAD)
2531 trace::printf ("%s @%p %s\n", __func__, this, this->name ());
2532#endif
2533 if (attr.th_stack_address != nullptr
2534 && attr.th_stack_size_bytes > stack::min_size ())
2535 {
2536 internal_construct_ (function, args, attr, nullptr, 0);
2537 }
2538 else
2539 {
2540 allocator_ = &allocator;
2541
2542 if (attr.th_stack_size_bytes > stack::min_size ())
2543 {
2544 allocated_stack_size_elements_ = (attr.th_stack_size_bytes
2545 + sizeof(typename allocator_type::value_type) - 1)
2546 / sizeof(typename allocator_type::value_type);
2547 }
2548 else
2549 {
2550 allocated_stack_size_elements_ = (stack::default_size ()
2551 + sizeof(typename allocator_type::value_type) - 1)
2552 / sizeof(typename allocator_type::value_type);
2553 }
2554
2555 // The reinterpret_cast<> is required since the allocator
2556 // uses allocation_element_t, which is usually larger.
2557 allocated_stack_address_ =
2558 reinterpret_cast<stack::element_t*> ((const_cast<allocator_type&> (allocator)).allocate (
2559 allocated_stack_size_elements_));
2560
2561 assert (allocated_stack_address_ != nullptr);
2562
2563 internal_construct_ (
2564 function,
2565 args,
2566 attr,
2567 allocated_stack_address_,
2568 allocated_stack_size_elements_
2569 * sizeof(typename allocator_type::value_type));
2570 }
