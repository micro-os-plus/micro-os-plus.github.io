154 template<typename T>
155 void
156 device_registry<T>::link (value_type* device)
157 {
158#if defined(DEBUG)
159 for (auto&& d : registry_list__)
160 {
161 // Validate the device name by checking duplicates.
162 if (std::strcmp (device->name (), d.name ()) == 0)
163 {
164 trace::puts ("Duplicate device name. Abort.");
165 std::abort ();
166 }
167 }
168#endif // DEBUG
169
170 registry_list__.link (*device);
171
172 trace::printf ("Device '%s%s' linked.\n", value_type::device_prefix (),
173 device->name ());
174 }
175
179 template<typename T>
180 T*
181 device_registry<T>::identify_device (const char* path)
182 {
183 assert(path != nullptr);
184
185 auto prefix = value_type::device_prefix ();
186 if (std::strncmp (prefix, path, std::strlen (prefix)) != 0)
187 {
188 // The device prefix does not match, not a device.
189 return nullptr;
190 }
191
192 // The prefix was identified; try to match the rest of the path.
193 auto name = path + std::strlen (prefix);
194
195 for (auto&& p : registry_list__)
196 {
197 if (p.match_name (name))
198 {
199 return static_cast<value_type*> (&p);
200 }
201 }
202
203 // Not a known device.
204 return nullptr;
205 }
