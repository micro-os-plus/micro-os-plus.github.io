179  template<typename T>
180  T*
181  device_registry<T>::identify_device (const char* path)
182  {
183  assert(path != nullptr);
184 
185  auto prefix = value_type::device_prefix ();
186  if (std::strncmp (prefix, path, std::strlen (prefix)) != 0)
187  {
188  // The device prefix does not match, not a device.
189  return nullptr;
190  }
191 
192  // The prefix was identified; try to match the rest of the path.
193  auto name = path + std::strlen (prefix);
194 
195  for (auto&& p : registry_list__)
196  {
197  if (p.match_name (name))
198  {
199  return static_cast<value_type*> (&p);
200  }
201  }
202 
203  // Not a known device.
204  return nullptr;
205  }
