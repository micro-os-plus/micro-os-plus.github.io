483 template<typename T>
484 T*
485 net_stack::allocate_socket (void)
486 {
487 using socket_type = T;
488
489 socket_type* sock;
490
491 if (deferred_sockets_list_.empty ())
492 {
493 sock = new socket_type (*this);
494 }
495 else
496 {
497 sock =
498 static_cast<socket_type*> (deferred_sockets_list_.unlink_head ());
499
500 // Call the constructor before reusing the object,
501 sock->~socket_type ();
502
503 // Placement new, run only the constructor.
504 new (sock) socket_type (*this);
505
506 // Deallocate all remaining elements in the list.
507 while (!deferred_sockets_list_.empty ())
508 {
509 socket_type* s =
510 static_cast<socket_type*> (deferred_sockets_list_.unlink_head ());
511
512 // Call the destructor and the deallocator.
513 delete s;
514 }
515 }
516 return sock;
517 }
518
519 template<typename T, typename L>
520 T*
521 net_stack::allocate_socket (L& locker)
522 {
523 using socket_type = T;
524
525 socket_type* sock;
526
527 if (deferred_sockets_list_.empty ())
528 {
529 sock = new socket_type (*this, locker);
530 }
531 else
532 {
533 sock =
534 static_cast<socket_type*> (deferred_sockets_list_.unlink_head ());
535
536 // Call the constructor before reusing the object,
537 sock->~socket_type ();
538
539 // Placement new, run only the constructor.
540 new (sock) socket_type (*this, locker);
541
542 // Deallocate all remaining elements in the list.
543 while (!deferred_sockets_list_.empty ())
544 {
545 socket_type* s =
546 static_cast<socket_type*> (deferred_sockets_list_.unlink_head ());
547
548 // Call the destructor and the deallocator.
549 delete s;
550 }
551 }
552 return sock;
