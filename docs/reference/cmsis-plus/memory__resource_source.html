214 template<typename T>
215 inline typename polymorphic_allocator<T>::value_type*
216 polymorphic_allocator<T>::allocate (std::size_t n)
217 {
218 if (n > max_size ())
219 {
220 estd::__throw_system_error (
221 EINVAL, "polymorphic_allocator<T>::allocate(std::size_t n)"
222 " 'n' exceeds maximum supported size");
223 }
224
225 return static_cast<value_type*> (res_->allocate (
226 n * sizeof(value_type), alignof(value_type)));
227 }

231 polymorphic_allocator<T>::deallocate (value_type* p, std::size_t n) noexcept
232 {
