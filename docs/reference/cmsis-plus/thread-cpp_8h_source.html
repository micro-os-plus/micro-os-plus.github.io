57 void
58 thread::delete_system_thread (void)
59 {

61  {
62  void* args = id_.native_thread_->function_args ();
63  if (args != nullptr && function_object_deleter_ != nullptr)
64  {
65  // Manually delete the function object used to store arguments.
66  function_object_deleter_ (args);
67  }
68 
69  // Manually delete the system thread.

71  }
72 }
