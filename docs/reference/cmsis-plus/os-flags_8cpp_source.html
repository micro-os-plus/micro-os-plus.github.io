42 result_t
43 event_flags::raise (flags::mask_t mask, flags::mask_t* oflags)
44 {
45 os_assert_err(mask != 0, EINVAL);
46
47 assert(port::interrupts::is_priority_valid ());
48
49 {
50 // ----- Enter critical section -------------------------------------
51 interrupts::critical_section ics;
52
53 if (oflags != nullptr)
54 {
55 *oflags = flags_mask_;
56 }
57
58 flags_mask_ |= mask;
59
60 // ----- Exit critical section --------------------------------------
61 }

65 bool
66 event_flags::check_raised (flags::mask_t mask, flags::mask_t* oflags,
67 flags::mode_t mode)
68 {
69 if (mask == flags::any)
70 {
71 // Any flag will do it.
72 if (flags_mask_ != 0)
73 {
74 if (oflags != nullptr)
75 {
76 *oflags = flags_mask_;
77 }
78
79 if (mode & flags::mode::clear)
80 {
81 // Clear them all.
82 flags_mask_ = 0;
83 }
84 return true;
85 }
86 }
87 else if (((((mode & flags::mode::all) != 0))
88 && ((flags_mask_ & mask) == mask))
89 || (((mode & flags::mode::any) != 0) && ((flags_mask_ & mask) != 0)))
90 {
91 if (oflags != nullptr)
92 {
93 *oflags = (flags_mask_ & mask);
94 }
95
96 if (mode & flags::mode::clear)
97 {
98 // Clear desired flags.
99 flags_mask_ &= ~mask;
100 }
101 return true;
102 }
103
104 return false;
105 }

107 flags::mask_t
108 event_flags::get (flags::mask_t mask, flags::mode_t mode)
109 {
110 flags::mask_t ret;
111 {
112 // ----- Enter critical section -------------------------------------
113 interrupts::critical_section ics;
114
115 if (mask == 0)
116 {
117 // Return the entire mask.
118 ret = flags_mask_;
119 }
120 else
121 {
122 ret = flags_mask_ & mask;
123 if ((mode & flags::mode::clear) != 0)
124 {
125 // Clear the selected bits; leave the rest untouched.
126 flags_mask_ &= ~mask;
127 }
128 }
129 // ----- Exit critical section --------------------------------------
130 }
131
132 // Return the selected bits.
133 return ret;
134 }
