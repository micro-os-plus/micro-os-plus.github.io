3593{

3595 {

3597 }

3601 return nullptr;
3602 }
3603

3605 attr.th_priority = thread_def->tpriority;
3606 attr.th_stack_size_bytes = thread_def->stacksize;
3607

3609 if (thread_def->tpriority >= osPriorityError)

3613

3615 for (uint32_t i = 0; i < thread_def->instances; ++i)
