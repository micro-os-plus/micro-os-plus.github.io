3596osThreadId
3597osThreadCreate (const osThreadDef_t* thread_def, void* args)
3598{
3599 if (interrupts::in_handler_mode ())
3600 {
3601 return nullptr;
3602 }
3603
3604 if (thread_def == nullptr)
3605 {
3606 return nullptr;
3607 }
3608
3609 thread::attributes attr;
3610 attr.th_priority = thread_def->tpriority;
3611 attr.th_stack_size_bytes = thread_def->stacksize;
3612
3613 // Creating thread with invalid priority should fail (validator requirement).
3614 if (thread_def->tpriority >= osPriorityError)
3615 {
3616 return nullptr;
3617 }
3618
3619 // Find a free slot in the tread definitions array.
3620 for (uint32_t i = 0; i < thread_def->instances; ++i)
3621 {
3622 thread* th = (thread*) &thread_def->data[i];
3623 if (th->state () == thread::state::undefined
3624 || th->state () == thread::state::destroyed)
3625 {
3626 if (attr.th_stack_size_bytes > 0)
3627 {
3628 attr.th_stack_address = &thread_def->stack[(i)
3629 * ((thread_def->stacksize + sizeof(uint64_t) - 1)
3630 / sizeof(uint64_t))];
3631 }
3632#pragma GCC diagnostic push
3633#pragma GCC diagnostic ignored "-Wcast-function-type"
3634 new (th) thread (thread_def->name,
3635 reinterpret_cast<thread::func_t> (thread_def->pthread), args, attr);
3636#pragma GCC diagnostic pop
3637
3638 // No need to yield here, already done by constructor.
3639 return reinterpret_cast<osThreadId> (th);
3640 }
3641 }
