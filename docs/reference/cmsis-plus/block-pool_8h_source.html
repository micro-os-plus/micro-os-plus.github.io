538 template<typename T, typename A>
539 block_pool_typed_allocated<T, A>::block_pool_typed_allocated (
540 const char* name, std::size_t blocks, const allocator_type& allocator) :
541 block_pool
542 { name }
543 {
544 trace::printf ("%s(%u,%p) @%p %s\n", __func__, blocks, &allocator, this,
545 this->name ());
546
547 // Remember the allocator, it'll be used by the destructor.
548 allocator_ =
549 static_cast<allocator_type*> (&const_cast<allocator_type&> (allocator));
550
551 void* addr = allocator_->allocate (blocks);
552 if (addr == nullptr)
553 {
554 estd::__throw_bad_alloc ();
555 }
556
557 internal_construct_ (blocks, sizeof(value_type), addr,
558 blocks * sizeof(value_type));
559 }
560
561 template<typename T, typename A>
562 block_pool_typed_allocated<T, A>::~block_pool_typed_allocated ()
563 {
564 trace::printf ("%s() @%p %s\n", __func__, this, this->name ());
565
566 // Skip in case a derived class did the deallocation.
567 if (allocator_ != nullptr)
568 {
569 allocator_->deallocate (
570 static_cast<typename allocator_traits::pointer> (pool_addr_),
571 blocks_);
572
573 // Prevent another deallocation.
574 allocator_ = nullptr;
575 }
576 }
