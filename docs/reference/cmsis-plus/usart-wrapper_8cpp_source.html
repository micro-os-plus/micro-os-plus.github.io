89 return_t
90 usart_wrapper::do_power (Power state) noexcept
91 {
92 return_t status;
93
94 if (state == Power::full)
95 {
96 status = driver_->Initialize (c_cb_func_);
97 if (status != ARM_DRIVER_OK)
98 {
99 return status;
100 }
101 }
102
103 status = driver_->PowerControl (static_cast<ARM_POWER_STATE> (state));
104
105 if (state == Power::off)
106 {
107 driver_->Uninitialize ();
108 }
109
110 return status;
111 }

150 return_t
151 usart_wrapper::do_control (serial::control_t ctrl) noexcept
152 {
153 switch (ctrl)
154 {
155 case serial::Control::disable_tx:
156 case serial::Control::disable_rx:
157 case serial::Control::disable_break:
158 return driver_->Control (
159 ctrl - (serial::Control::disable_tx - serial::Control::enable_tx),
160 0);
161 }
162 return driver_->Control (ctrl, 1);
163 }
