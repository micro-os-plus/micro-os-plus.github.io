308  void
309  memory_pool::internal_construct_ (std::size_t blocks,
310  std::size_t block_size_bytes,
311  const attributes& attr,
312  void* pool_address,
313  std::size_t pool_size_bytes)
314  {
315  os_assert_throw(!interrupts::in_handler_mode (), EPERM);
316 
317 #if !defined(OS_USE_RTOS_PORT_MEMORY_POOL)
318  clock_ = attr.clock != nullptr ? attr.clock : &sysclock;
319 #endif
320 
321  blocks_ = static_cast<memory_pool::size_t> (blocks);
322  assert(blocks_ == blocks);
323  assert(blocks_ > 0);
324 
325  // Adjust block size to multiple of pointer.
326  // Blocks must be large enough to store a pointer, used
327  // to construct the list of free blocks.
328  block_size_bytes_ = (static_cast<memory_pool::size_t> (block_size_bytes
329  + __SIZEOF_POINTER__ - 1))
330  & (static_cast<memory_pool::size_t> (~(__SIZEOF_POINTER__ - 1)));
331 
332  // If the storage is given explicitly, override attributes.
333  if (pool_address != nullptr)
334  {
335  // The attributes should not define any storage in this case.
336  assert(attr.mp_pool_address == nullptr);
337 
338  pool_addr_ = pool_address;
339  pool_size_bytes_ = pool_size_bytes;
340  }
341  else
342  {
343  pool_addr_ = attr.mp_pool_address;
344  pool_size_bytes_ = attr.mp_pool_size_bytes;
345  }
346 
347  // Blocks must be pointer aligned.
348  void* p = pool_addr_;
349  std::size_t sz = pool_size_bytes_;
350  pool_addr_ = static_cast<char*> (std::align (__SIZEOF_POINTER__,
351  blocks_ * block_size_bytes_,
352  p, sz));
353 
354 #if defined(OS_TRACE_RTOS_MEMPOOL)
355  trace::printf ("%s() @%p %s %u %u %p %u\n", __func__, this, name (),
356  blocks_, block_size_bytes_, pool_addr_, pool_size_bytes_);
357 #endif
358 
359  std::size_t storage_size = compute_allocated_size_bytes<void*> (
360  blocks_, block_size_bytes_);
361 
362  if (pool_addr_ != nullptr)
363  {
364  os_assert_throw(pool_size_bytes_ > 0, EINVAL);
365  os_assert_throw(pool_size_bytes_ >= storage_size, EINVAL);
366  }
367 
368  os_assert_throw(pool_addr_ != nullptr, ENOMEM);
369 
370  internal_init_ ();

416  /*
417  * Construct the linked list of blocks and initialise the
418  * internal pointers and counters.
419  */
420  void
421  memory_pool::internal_init_ (void)
422  {
423  // Construct a linked list of blocks. Store the pointer at
424  // the beginning of each block. Each block
425  // will hold the address of the next free block, or nullptr at the end.
426  char* p = static_cast<char*> (pool_addr_);
427  for (std::size_t i = 1; i < blocks_; ++i)
428  {
429  // Compute the address of the next block;
430  char* pn = p + block_size_bytes_;
431 
432  // Make this block point to the next one.
433  *(static_cast<void**> (static_cast<void*> (p))) = pn;
434  // Advance pointer
435  p = pn;
436  }
437 
438  // Mark end of list.
439  *(static_cast<void**> (static_cast<void*> (p))) = nullptr;
440 
441  first_ = pool_addr_; // Pointer to first block.
442 
443  count_ = 0; // No allocated blocks.
444  }

754  memory_pool::free (void* block)
755  {
756 #if defined(OS_TRACE_RTOS_MEMPOOL)
757  trace::printf ("%s(%p) @%p %s\n", __func__, block, this, name ());
758 #endif
759 
760  assert(port::interrupts::is_priority_valid ());
761 
762  // Validate pointer.
763  if ((block < pool_addr_)
764  || (block
765  >= (static_cast<char*> (pool_addr_) + blocks_ * block_size_bytes_)))
766  {
767 #if defined(OS_TRACE_RTOS_MEMPOOL)
768  trace::printf ("%s(%p) EINVAL @%p %s\n", __func__, block, this,
769  name ());
770 #endif
771  return EINVAL;
772  }
773 
774  {
775  // ----- Enter critical section -------------------------------------
776  interrupts::critical_section ics;
777 
778  // Perform a push_front() on the single linked LIFO list,
779  // i.e. add the block to the beginning of the list.
780 
781  // Link previous list to this block; may be null, but it does
782  // not matter.
783  *(static_cast<void**> (block)) = first_;
784

786  first_ = block;
787 
788  --count_;

790  }
791 
792  // Wake-up one thread, if any.
793  list_.resume_one ();
