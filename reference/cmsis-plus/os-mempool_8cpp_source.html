308  void
309  memory_pool::internal_construct_ (std::size_t blocks,
310  std::size_t block_size_bytes,
311  const attributes& attr,
312  void* pool_address,
313  std::size_t pool_size_bytes)
314  {
315  // Don't call this from interrupt handlers.
316  os_assert_throw(!interrupts::in_handler_mode (), EPERM);
317 
318 #if !defined(OS_USE_RTOS_PORT_MEMORY_POOL)
319  clock_ = attr.clock != nullptr ? attr.clock : &sysclock;
320 #endif
321 
322  blocks_ = static_cast<memory_pool::size_t> (blocks);
323  assert(blocks_ == blocks);
324  assert(blocks_ > 0);
325 
326  // Adjust block size to multiple of pointer.
327  // Blocks must be large enough to store a pointer, used
328  // to construct the list of free blocks.
329  block_size_bytes_ = (static_cast<memory_pool::size_t> (block_size_bytes
330  + __SIZEOF_POINTER__ - 1))
331  & (static_cast<memory_pool::size_t> (~(__SIZEOF_POINTER__ - 1)));
332 
333  // If the storage is given explicitly, override attributes.
334  if (pool_address != nullptr)
335  {
336  // The attributes should not define any storage in this case.
337  assert(attr.mp_pool_address == nullptr);
338 
339  pool_addr_ = pool_address;
340  pool_size_bytes_ = pool_size_bytes;
341  }
342  else
343  {
344  pool_addr_ = attr.mp_pool_address;
345  pool_size_bytes_ = attr.mp_pool_size_bytes;
346  }
347 
348  // Blocks must be pointer aligned.
349  void* p = pool_addr_;
350  std::size_t sz = pool_size_bytes_;
351  pool_addr_ = static_cast<char*> (std::align (__SIZEOF_POINTER__,
352  blocks_ * block_size_bytes_,
353  p, sz));
354 
355 #if defined(OS_TRACE_RTOS_MEMPOOL)
356  trace::printf ("%s() @%p %s %u %u %p %u\n", __func__, this, name (),
357  blocks_, block_size_bytes_, pool_addr_, pool_size_bytes_);
358 #endif
359 
360  std::size_t storage_size = compute_allocated_size_bytes<void*> (
361  blocks_, block_size_bytes_);
362 
363  if (pool_addr_ != nullptr)
364  {
365  // The pool must be real, and have a non zero size.
366  os_assert_throw(pool_size_bytes_ > 0, EINVAL);
367  // The pool must fit the storage.
368  os_assert_throw(pool_size_bytes_ >= storage_size, EINVAL);
369  }
370 
371  // The pool must have a real address.
372  os_assert_throw(pool_addr_ != nullptr, ENOMEM);
373 
374  internal_init_ ();
375 
376  }

421  /*
422  * Construct the linked list of blocks and initialise the
423  * internal pointers and counters.
424  */
425  void
426  memory_pool::internal_init_ (void)
427  {
428  // Construct a linked list of blocks. Store the pointer at
429  // the beginning of each block. Each block
430  // will hold the address of the next free block, or nullptr at the end.
431  char* p = static_cast<char*> (pool_addr_);
432  for (std::size_t i = 1; i < blocks_; ++i)
433  {
434  // Compute the address of the next block;
435  char* pn = p + block_size_bytes_;
436 
437  // Make this block point to the next one.
438  *(static_cast<void**> (static_cast<void*> (p))) = pn;
439  // Advance pointer
440  p = pn;
441  }
442 
443  // Mark end of list.
444  *(static_cast<void**> (static_cast<void*> (p))) = nullptr;
445 
446  first_ = pool_addr_; // Pointer to first block.
447 
448  count_ = 0; // No allocated blocks.
449  }
