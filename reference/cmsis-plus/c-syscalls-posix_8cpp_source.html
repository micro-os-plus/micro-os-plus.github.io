53 // To facilitate testing on POSIX platforms, and also to allow
54 // integration on custom platforms, all function names are prefixed
55 // with '__posix_'.
56 // For embedded environments it is possible to also add aliases to the
57 // standard, non-prefixed names, by adding the following preprocessor
58 // definition: OS_INCLUDE_STANDARD_POSIX_FUNCTIONS

60 // Notes: Reentrancy and 'errno'.
61 //
62 // The standard headers define errno as '*(__errno())';
63 // If you use a multi-threaded environment, be sure you
64 // redefine __errno() to return a thread specific pointer.

89 int
90 __posix_open (const char* path, int oflag, ...)
91 {
92  va_list args;
93  va_start(args, oflag);
94  auto* const io = posix::vopen (path, oflag, args);
95  va_end(args);
96 
97  if (io == nullptr)
98  {
99  // Return POSIX style error indicator.
100  return -1;
101  }
102 
103  // Return non-negative POSIX file descriptor.
104  return io->file_descriptor ();
105 }

107 int
108 __posix_close (int fildes)
109 {
110  // The flow is identical for all POSIX functions: identify the C++
111  // object and call the corresponding C++ method.
112  auto* const io = posix::file_descriptors_manager::io (fildes);
113  if (io == nullptr)
114  {
115  errno = EBADF;
116  return -1;
117  }
118  return io->close ();
119 }

169 int
170 __posix_ioctl (int fildes, int request, ...)
171 {
172  auto* const io = posix::file_descriptors_manager::io (fildes);
173  if (io == nullptr)
174  {
175  errno = EBADF;
176  return -1;
177  }
178 
179  // Works only on STREAMS (CherDevices, in this implementation)
180  if ((io->get_type () & posix::io::type::char_device) == 0)
181  {
182  errno = ENOTTY; // Not a stream.
183  return -1;
184  }
185 
186  va_list args;
187  va_start(args, request);
188  int ret = (static_cast<posix::char_device*> (io))->vioctl (request, args);
189  va_end(args);
190 
191  return ret;
192 }
