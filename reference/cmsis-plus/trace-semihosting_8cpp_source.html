57  // Semihosting is another output channel that can be used for the trace
58  // messages. It comes in two flavours: STDOUT and DEBUG. The STDOUT channel
59  // is the equivalent of the stdout in POSIX and in most cases it is
60  // forwarded to the GDB server stdout stream. The debug channel is a
61  // separate channel. STDOUT is buffered, so nothing is displayed until
62  // a \n; DEBUG is not buffered, but can be slow.
63  //
64  // Choosing between semihosting stdout and debug depends on the capabilities
65  // of your GDB server, and also on specific needs. It is recommended to test
66  // DEBUG first, and if too slow, try STDOUT.
67  //
68  // The JLink GDB server fully support semihosting, and both configurations
69  // are available; to activate it, use "monitor semihosting enable" or check
70  // the corresponding button in the JLink Debugging plug-in.
71  // In OpenOCD, support for semihosting can be enabled using
72  // "monitor arm semihosting enable".
73  //
74  // Note: Applications built with semihosting output active normally cannot
75  // be executed without the debugger connected and active, since they use
76  // BKPT to communicate with the host. However, with a carefully written
77  // HardFault_Handler, the semihosting BKPT calls can be processed, making
78  // possible to run semihosting applications as standalone, without being
79  // terminated with hardware faults.

89  ssize_t
90  write (const void* buf, std::size_t nbyte)
91  {
92  if (buf == nullptr || nbyte == 0)
93  {
94  return 0;
95  }
96 
97  const char* cbuf = (const char*) buf;
98 
99  // Since the single character debug channel is quite slow, try to
100  // optimise and send a null terminated string, if possible.
101  if (cbuf[nbyte] == '\0')
102  {
103  // send string
104  call_host (SEMIHOSTING_SYS_WRITE0, (void*) cbuf);
105  }
106  else
107  {
108  // If not, use a local buffer to speed things up.
109  // For re-entrance, this bugger must be allocated on the stack,
110  // so be cautious with the size.
111  char tmp[OS_INTEGER_TRACE_SEMIHOSTING_TMP_ARRAY_SIZE];
112  size_t togo = nbyte;
113  while (togo > 0)
114  {
115  std::size_t n = ((togo < sizeof(tmp)) ? togo : sizeof(tmp));
116  std::size_t i = 0;
117  for (; i < n; ++i, ++cbuf)
118  {
119  tmp[i] = *cbuf;
120  }
121  tmp[i] = '\0';
122 
123  call_host (SEMIHOSTING_SYS_WRITE0, (void*) tmp);
124 
125  togo -= n;
126  }
127  }
128 
129  // All bytes written.
130  return (ssize_t) nbyte;
131  }
132

135  ssize_t
136  write (const void* buf, std::size_t nbyte)
137  {
138  if (buf == nullptr || nbyte == 0)
139  {
140  return 0;
141  }
142 
143  static int handle; // STATIC!
144 
145  void* block[3];
146  int ret;
147 
148  if (handle == 0)
149  {
150  // On the very first call get the file handle from the host.
151 
152  // Special filename for stdin/out/err.
153  block[0] = (void*) ":tt";
154  block[1] = (void*) 4;// mode "w"
155  // Length of ":tt", except null terminator.
156  block[2] = (void*) (sizeof(":tt") - 1);
157 
158  ret = call_host (SEMIHOSTING_SYS_OPEN, (void*) block);
159  if (ret == -1)
160  {
161  return -1;
162  }
163 
164  handle = ret;
165  }
166 
167  block[0] = (void*) handle;
168  block[1] = (void*) buf;
169  block[2] = (void*) nbyte;
170  // Send character array to host file/device.
171  ret = call_host (SEMIHOSTING_SYS_WRITE, (void*) block);
172  // This call returns the number of bytes NOT written (0 if all ok).
173 
174  // -1 is not a legal value, but SEGGER seems to return it
175  if (ret == -1)
176  {
177  return -1;
178  }
179 
180  // The compliant way of returning errors.
181  if (ret == (int) nbyte)
182  {
183  return -1;
184  }
185 
186  // Return the number of bytes written.
187  return (ssize_t) (nbyte) - (ssize_t) ret;
188  }
