228  template<typename Rep_T, typename Period_T>
229  bool
230  timed_mutex::try_lock_for (
231  const std::chrono::duration<Rep_T, Period_T>& rel_time)
232  {
233  using namespace std::chrono;
234  os::rtos::clock::duration_t ticks = 0;
235  if (rel_time > duration<Rep_T, Period_T>::zero ())
236  {
237  ticks =
238  static_cast<os::rtos::clock::duration_t> (os::estd::chrono::ceil<
239  os::estd::chrono::systicks> (rel_time).count ());
240  }
241 
242  os::rtos::result_t res;
243  res = nm_.timed_lock (ticks);
244  if (res == os::rtos::result::ok)
245  {
246  return true;
247  }
248  else if (res == ETIMEDOUT)
249  {
250  return false;
251  }
252 
253  os::estd::__throw_system_error (static_cast<int> (res),
254  "timed_mutex try_lock failed");
255  return false;
256  }

258  template<typename Clock_T, typename Duration_T>
259  bool
260  timed_mutex::try_lock_until (
261  const std::chrono::time_point<Clock_T, Duration_T>& abs_time)
262  {
263  using clock = Clock_T;
264 
265  auto now = clock::now ();
266  while (now < abs_time)
267  {
268  if (try_lock_for (abs_time - now))
269  {
270  return true;
271  }
272  now = clock::now ();
273  }
274 
275  return false;
276  }
