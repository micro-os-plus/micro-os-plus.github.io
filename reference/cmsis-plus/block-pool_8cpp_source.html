144  block_pool::internal_construct_ (std::size_t blocks,
145  std::size_t block_size_bytes, void* addr,
146  std::size_t bytes) noexcept
147  {
148  blocks_ = blocks;
149 
150  block_size_bytes_ = rtos::memory::align_size (block_size_bytes,

152  assert(block_size_bytes_ >= sizeof(void*));

154  assert(addr != nullptr);

156 
157  std::size_t align_sz = bytes;

181  void
182  block_pool::internal_reset_ (void) noexcept
183  {
184  // Construct a linked list of blocks. Store the pointer at
185  // the beginning of each block. Each block
186  // will hold the address of the next free block, or nullptr at the end.
187  char* p = static_cast<char*> (pool_addr_);
188  for (std::size_t i = 1; i < blocks_; ++i)
189  {
190  // Compute the address of the next block;
191  char* pn = p + block_size_bytes_;
192 
193  // Make this block point to the next one.
194  *(static_cast<void**> (static_cast<void*> (p))) = pn;
195  // Advance pointer
196  p = pn;
197  }
198 
199  // Mark end of list.
200  *(static_cast<void**> (static_cast<void*> (p))) = nullptr;
201 
202  first_ = pool_addr_; // Pointer to first block.
203 
204  count_ = 0; // No allocated blocks.
205 
206  allocated_bytes_ = 0;
207  max_allocated_bytes_ = 0;
208  free_bytes_ = total_bytes_;
209  allocated_chunks_ = 0;
210  free_chunks_ = blocks_;
211 
212  }
