396  result_t
397  semaphore::wait ()
398  {
399 #if defined(OS_TRACE_RTOS_SEMAPHORE)
400  trace::printf ("%s() @%p %s <%u\n", __func__, this, name (), count_);
401 #endif
402 
403  os_assert_err(!interrupts::in_handler_mode (), EPERM);
404  os_assert_err(!scheduler::locked (), EPERM);
405 
406 #if defined(OS_USE_RTOS_PORT_SEMAPHORE)
407 
408  return port::semaphore::wait (this);
409 
410 #else
411 
412  // Extra test before entering the loop, with its inherent weight.
413  // Trade size for speed.
414  {
415  // ----- Enter critical section -------------------------------------
416  interrupts::critical_section ics;
417 
418  if (internal_try_wait_ ())
419  {
420  return result::ok;
421  }
422  // ----- Exit critical section --------------------------------------
423  }
424 
425  thread& crt_thread = this_thread::thread ();
426 
427  // Prepare a list node pointing to the current thread.
428  // Do not worry for being on stack, it is temporarily linked to the
429  // list and guaranteed to be removed before this function returns.
430  internal::waiting_thread_node node
431  { crt_thread };
432 
433  for (;;)
434  {
435  {
436  // ----- Enter critical section ---------------------------------
437  interrupts::critical_section ics;
438 
439  if (internal_try_wait_ ())
440  {
441  return result::ok;
442  }
443 
444  // Add this thread to the semaphore waiting list.
445  scheduler::internal_link_node (list_, node);
446  // state::suspended set in above link().
447  // ----- Exit critical section ----------------------------------
448  }
449 
450  port::scheduler::reschedule ();
451 
452  // Remove the thread from the semaphore waiting list,
453  // if not already removed by post().
454  scheduler::internal_unlink_node (node);
455 
456  if (crt_thread.interrupted ())
457  {
458 #if defined(OS_TRACE_RTOS_SEMAPHORE)
459  trace::printf ("%s() EINTR @%p %s\n", __func__, this, name ());
460 #endif
461  return EINTR;
462  }
463  }
464 
465  /* NOTREACHED */
466  return ENOTRECOVERABLE;
467 
468 #endif
469  }

561  semaphore::timed_wait (clock::duration_t timeout)
562  {
563 #if defined(OS_TRACE_RTOS_SEMAPHORE)
564  trace::printf ("%s(%u) @%p %s <%u\n", __func__,
565  static_cast<unsigned int> (timeout), this, name (),
566  count_);
567 #endif
568 
569  os_assert_err(!interrupts::in_handler_mode (), EPERM);
570  os_assert_err(!scheduler::locked (), EPERM);
571 
572 #if defined(OS_USE_RTOS_PORT_SEMAPHORE)
573 
574  return port::semaphore::timed_wait (this, timeout);
575 
576 #else
577 
578  // Extra test before entering the loop, with its inherent weight.
579  // Trade size for speed.
580  {
581  // ----- Enter critical section -------------------------------------
582  interrupts::critical_section ics;
583 
584  if (internal_try_wait_ ())
585  {
586  return result::ok;
587  }
588  // ----- Exit critical section --------------------------------------
589  }
590 
591  thread& crt_thread = this_thread::thread ();
592 
593  // Prepare a list node pointing to the current thread.
594  // Do not worry for being on stack, it is temporarily linked to the
595  // list and guaranteed to be removed before this function returns.
596  internal::waiting_thread_node node
597  { crt_thread };
598 
599  internal::clock_timestamps_list& clock_list = clock_->steady_list ();
600  clock::timestamp_t timeout_timestamp = clock_->steady_now () + timeout;
601 
602  // Prepare a timeout node pointing to the current thread.
603  internal::timeout_thread_node timeout_node
604  { timeout_timestamp, crt_thread };
605 
606  for (;;)
607  {
608  {
609  // ----- Enter critical section ---------------------------------
610  interrupts::critical_section ics;
611 
612  if (internal_try_wait_ ())
613  {
614  return result::ok;
615  }
616 
617  // Add this thread to the semaphore waiting list,
618  // and the clock timeout list.
619  scheduler::internal_link_node (list_, node, clock_list,
620  timeout_node);
621  // state::suspended set in above link().
622  // ----- Exit critical section ----------------------------------
623  }
624 
625  port::scheduler::reschedule ();
626 
627  // Remove the thread from the semaphore waiting list,
628  // if not already removed by post() and from the clock
629  // timeout list, if not already removed by the timer.
630  scheduler::internal_unlink_node (node, timeout_node);
631 
632  if (crt_thread.interrupted ())
633  {
634 #if defined(OS_TRACE_RTOS_SEMAPHORE)
635  trace::printf ("%s(%u) EINTR @%p %s\n", __func__,
636  static_cast<unsigned int> (timeout), this,
637  name ());
638 #endif
639  return EINTR;
640  }
641 
642  if (clock_->steady_now () >= timeout_timestamp)
643  {
644 #if defined(OS_TRACE_RTOS_SEMAPHORE)
645  trace::printf ("%s(%u) ETIMEDOUT @%p %s\n", __func__,
646  static_cast<unsigned int> (timeout), this,
647  name ());
648 #endif
649  return ETIMEDOUT;
650  }
651  }
652 
653  /* NOTREACHED */
654  return ENOTRECOVERABLE;
655 
656 #endif
657  }
