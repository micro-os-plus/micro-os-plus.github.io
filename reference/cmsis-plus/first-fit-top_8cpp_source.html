128  void*
129  first_fit_top::do_allocate (std::size_t bytes, std::size_t alignment)
130  {
131  // TODO: consider `alignment` if > block_align.
132 
133  std::size_t alloc_size = rtos::memory::align_size (bytes, chunk_align);
134  alloc_size += block_padding;
135  alloc_size += chunk_offset;
136 
137  alloc_size = os::rtos::memory::max (alloc_size, block_minchunk);
138 
139  chunk_t* chunk;
140 
141  while (true)
142  {
143  chunk_t* prev_chunk = free_list_;
144  chunk = prev_chunk;
145 
146  while (chunk)
147  {
148  int rem = static_cast<int> (chunk->size - alloc_size);
149  if (rem >= 0)
150  {
151  if ((static_cast<std::size_t> (rem)) >= block_minchunk)
152  {
153  // Found a chunk that is much larger than required size
154  // (at least one more chunk is available);
155  // break it into two chunks and return the second one.
156 
157  chunk->size = static_cast<std::size_t> (rem);
158  chunk =
159  reinterpret_cast<chunk_t *> (reinterpret_cast<char *> (chunk)
160  + rem);
161  chunk->size = alloc_size;
162 
163  // Splitting one chunk creates one more chunk.
164  ++free_chunks_;
165  }
166  else
167  {
168  // Found a chunk that is exactly the size or slightly
169  // larger than the requested size; return this chunk.
170 
171  if (prev_chunk == chunk)
172  {
173  // This implies p==r==free_list, i.e. the list head.
174  // The next chunk becomes the first list element.
175  free_list_ = chunk->next;
176 
177  // If this was the last chunk, the free list is empty.
178  }
179  else
180  {
181  // Normal case. Remove it from the free_list.
182  prev_chunk->next = chunk->next;
183  }
184  }
185  break;
186  }
187  prev_chunk = chunk;
188  chunk = chunk->next;
189  }
190 
191  if (chunk != nullptr)
192  {
193  break;
194  }
195 
196  if (out_of_memory_handler_ == nullptr)
197  {
198 #if defined(OS_TRACE_LIBCPP_MEMORY_RESOURCE)
199  trace::printf ("%s(%u,%u)=0 @%p %s\n", __func__, bytes, alignment,
200  this, name ());
201 #endif
202 
203  return nullptr;
204  }
205 
206 #if defined(OS_TRACE_LIBCPP_MEMORY_RESOURCE)
207  trace::printf ("%s(%u,%u) @%p %s out of memory\n", __func__, bytes,
208  alignment, this, name ());
209 #endif
210  out_of_memory_handler_ ();
211 
212  // If the handler returned, assume it freed some memory
213  // and try again to allocate.
214  }
215 
216  // Update statistics.
217  // What is subtracted from free is added to allocated.
218  internal_increase_allocated_statistics (chunk->size);
219 
220  // Compute pointer to payload area.
221  char* payload = reinterpret_cast<char *> (chunk) + chunk_offset;
222 
223  // Align it to block_align.
224  void* aligned_payload = payload;
225  std::size_t aligned_size = chunk->size - chunk_offset;
226 
227  void* res;
228  res = std::align (block_align, bytes, aligned_payload, aligned_size);
229  if (res != nullptr)
230  {
231  assert(res != nullptr);
232  }
233 
234  // Compute the possible alignment offset.
235  std::ptrdiff_t offset = static_cast<char *> (aligned_payload) - payload;
236  if (offset)
237  {
238  // If non-zero, store it in the gap left by alignment in the
239  // chunk header.
240 
241  chunk_t* adj_chunk =
242  reinterpret_cast<chunk_t *> (static_cast<char *> (aligned_payload)
243  - chunk_offset);
244  adj_chunk->size = static_cast<std::size_t> (-offset);
245  }
246 
247 #if defined(OS_TRACE_LIBCPP_MEMORY_RESOURCE)
248  trace::printf ("%s(%u,%u)=%p,%u @%p %s\n", __func__, bytes, alignment,
249  aligned_payload, alloc_size, this, name ());
250 #endif
251 
252  return aligned_payload;
253  }

270  void
271  first_fit_top::do_deallocate (void* addr, std::size_t bytes,
272  std::size_t alignment) noexcept
273  {
274 #if defined(OS_TRACE_LIBCPP_MEMORY_RESOURCE)
275  trace::printf ("%s(%p,%u,%u) @%p %s\n", __func__, addr, bytes, alignment,
276  this, name ());
277 #endif
278 
279  // The address must be inside the arena; no exceptions.
280  if ((addr < arena_addr_)
281  || (addr > (static_cast<char*> (arena_addr_) + total_bytes_)))
282  {
283  assert(false);
284  return;
285  }
286 
287  // Compute the chunk address from the user address.
288  chunk_t* chunk = reinterpret_cast<chunk_t *> (static_cast<char *> (addr)
289  - chunk_offset);
290 
291  // If the block was aligned, the offset appears as size; adjust back.
292  if (static_cast<std::ptrdiff_t> (chunk->size) < 0)
293  {
294  chunk = reinterpret_cast<chunk_t *> (reinterpret_cast<char *> (chunk)
295  + static_cast<std::ptrdiff_t> (chunk->size));
296  }
297 
298  if (bytes)
299  {
300  // If size is known, validate.
301  // (when called from free(), the size is not known).
302  if (bytes + chunk_offset > chunk->size)
303  {
304  assert(false);
305  return;
306  }
307  }
308 
309  // Update statistics.
310  // What is subtracted from allocated is added to free.
311  internal_decrease_allocated_statistics (chunk->size);
312 
313  // If the free list is empty, create it with the current chunk, alone.
314  if (free_list_ == nullptr)
315  {
316  // Mark the end of the list with a null pointer.
317  chunk->next = nullptr;
318 
319  // The chunk becomes the first list element.
320  free_list_ = chunk;
321  assert(free_chunks_ == 1);
322 
323  return;
324  }
325 
326  // The free list exists; is the chunk before the list head?
327  if (chunk < free_list_)
328  {
329  // Is the chunk *right* before the list head?
330  if (reinterpret_cast<char *> (chunk) + chunk->size
331  == reinterpret_cast<char *> (free_list_))
332  {
333  // Coalesce chunk to the list head.
334  chunk->size += free_list_->size;
335  chunk->next = free_list_->next; // May be nullptr
336 
337  // Coalescing means one less chunk.
338  --free_chunks_;
339  }
340  else
341  {
342  // Insert before the list head.
343  chunk->next = free_list_;
344  }
345  // The chunk becomes the new list head.
346  free_list_ = chunk;
347 
348  return;
349  }
350 
351  // Walk the free list to find the place to insert,
352  // (the list must be ordered by addresses).
353  // Warning: not deterministic!
354 
355  chunk_t* next_chunk = free_list_;
356  chunk_t* prev_chunk;
357  do
358  {
359  prev_chunk = next_chunk;
360  next_chunk = next_chunk->next;
361  }
362  while (next_chunk != nullptr && next_chunk <= chunk);
363 
364  // Now prev_chunk <= chunk and either next_chunk == nullptr or
365  // next_chunk > chunk.
366  // Try to merge with chunks immediately before/after it.
367 
368  if (reinterpret_cast<char *> (prev_chunk) + prev_chunk->size
369  == reinterpret_cast<char *> (chunk))
370  {
371  // Chunk to be freed is adjacent to a free chunk before it.
372  prev_chunk->size += chunk->size;
373 
374  // Coalescing means one less chunk.
375  --free_chunks_;
376 
377  // If the merged chunk is also adjacent to the chunk after it,
378  // merge again. Does not match if next_chunk == nullptr.
379  if (reinterpret_cast<char *> (prev_chunk) + prev_chunk->size
380  == reinterpret_cast<char *> (next_chunk))
381  {
382  prev_chunk->size += next_chunk->size;
383  prev_chunk->next = next_chunk->next;
384 
385  // Coalescing means one less chunk.
386  --free_chunks_;
387  }
388  }
389  else if (reinterpret_cast<char *> (prev_chunk) + prev_chunk->size
390  > reinterpret_cast<char *> (chunk))
391  {
392  // Already freed.
393 
394  // Revert statistics.
395  // What is subtracted from free is added to allocated.
396  allocated_bytes_ += chunk->size;
397  free_bytes_ -= chunk->size;
398  ++allocated_chunks_;
399  --free_chunks_;
400 
401  trace::printf ("%s(%p,%u,%u) @%p %s already freed\n", __func__, addr,
402  bytes, alignment, this, name ());
403 
404  return;
405  }
406  // Does not match if next_chunk == nullptr.
407  else if (reinterpret_cast<char *> (chunk) + chunk->size
408  == reinterpret_cast<char *> (next_chunk))
409  {
410  // The chunk to be freed is adjacent to a free chunk after it.
411  chunk->size += next_chunk->size;
412  chunk->next = next_chunk->next; // May be nullptr.
413  prev_chunk->next = chunk;
414 
415  // Coalescing means one less chunk.
416  --free_chunks_;
417  }
418  else
419  {
420  // Not adjacent to any chunk. Just insert it.
421  // The result is a new fragment. Not great...
422  chunk->next = next_chunk; // May be nullptr.
423  prev_chunk->next = chunk;
424  }
425  }
