73  {

91  }

94  || (((mode & flags::mode::any) != 0) && ((flags_mask_ & mask) != 0)))
95  {
96  if (oflags != nullptr)
97  {
98  *oflags = (flags_mask_ & mask);
99  }
100 
101  if (mode & flags::mode::clear)
102  {
103  // Clear desired flags.
104  flags_mask_ &= ~mask;
105  }
106  return true;
107  }

112  flags::mask_t
113  event_flags::get (flags::mask_t mask, flags::mode_t mode)
114  {
115  flags::mask_t ret;
116  {
117  // ----- Enter critical section -------------------------------------
118  interrupts::critical_section ics;
119 
120  if (mask == 0)
121  {
122  // Return the entire mask.
123  ret = flags_mask_;
124  }
125  else
126  {
127  ret = flags_mask_ & mask;
128  if ((mode & flags::mode::clear) != 0)
129  {
130  // Clear the selected bits; leave the rest untouched.
131  flags_mask_ &= ~mask;
132  }
133  }
134  // ----- Exit critical section --------------------------------------
135  }
136 
137  // Return the selected bits.
138  return ret;
139  }
