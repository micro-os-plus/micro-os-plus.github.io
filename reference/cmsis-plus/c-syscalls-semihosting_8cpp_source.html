92 /*
93  * User file descriptors (fd) are integer indexes into
94  * the openfiles[] array. Error checking is done by using
95  * findslot().
96  *
97  * This openfiles array is manipulated directly by only
98  * these 5 functions:
99  *
100  * findslot() - Translate entry.
101  * newslot() - Find empty entry.
102  * initilise_monitor_handles() - Initialize entries.
103  * _swiopen() - Initialize entry.
104  * _close() - Handle stdout == stderr case.
105  *
106  * Every other function must use findslot().
107  */
108

110 #define OS_INTEGER_SEMIHOSTING_MAX_OPEN_FILES (20)
111 #endif

118 // Return a pointer to the structure associated with
119 // the user file descriptor fd.
120 static struct fdent*
121 __semihosting_findslot (int fd)
122 {
123  // User file descriptor is out of range.
124  if ((unsigned int) fd >= OS_INTEGER_SEMIHOSTING_MAX_OPEN_FILES)
125  {
126  return nullptr;
127  }
128 
129  // User file descriptor is open?
130  if (openfiles[fd].handle == -1)
131  {
132  return nullptr;
133  }
134 
135  // Valid.
136  return &openfiles[fd];
137 }
138 
139 // Return the next lowest numbered free file
140 // structure, or -1 if we can't find one.
141 static int
142 __semihosting_newslot (void)
143 {
144  int i;
145  for (i = 0; i < OS_INTEGER_SEMIHOSTING_MAX_OPEN_FILES; i++)
146  {
147  if (openfiles[i].handle == -1)
148  {
149  break;
150  }
151  }
152 
153  if (i == OS_INTEGER_SEMIHOSTING_MAX_OPEN_FILES)
154  {
155  return -1;
156  }
157 
158  return i;
159 }

187 /* fd, is a user file descriptor. */
188 static int
189 __semihosting_lseek (int fd, int ptr, int dir)
190 {
191  struct fdent *pfd;
192 
193  /* Valid file descriptor? */
194  pfd = __semihosting_findslot (fd);
195  if (pfd == NULL)
196  {
197  errno = EBADF;
198  return -1;
199  }
200 
201  /* Valid whence? */
202  if ((dir != SEEK_CUR) && (dir != SEEK_SET) && (dir != SEEK_END))
203  {
204  errno = EINVAL;
205  return -1;
206  }
207 
208  /* Convert SEEK_CUR to SEEK_SET */
209  if (dir == SEEK_CUR)
210  {
211  ptr = pfd->pos + ptr;
212  /* The resulting file offset would be negative. */
213  if (ptr < 0)
214  {
215  errno = EINVAL;
216  if ((pfd->pos > 0) && (ptr > 0))
217  {
218  errno = EOVERFLOW;
219  }
220  return -1;
221  }
222  dir = SEEK_SET;
223  }
224 
225  int block[2];
226  int res;
227 
228  if (dir == SEEK_END)
229  {
230  block[0] = pfd->handle;
231  res = __semihosting_checkerror (call_host (SEMIHOSTING_SYS_FLEN, block));
232  if (res == -1)
233  {
234  return -1;
235  }
236  ptr += res;
237  }
238 
239  /* This code only does absolute seeks. */
240  block[0] = pfd->handle;
241  block[1] = ptr;
242  res = __semihosting_checkerror (call_host (SEMIHOSTING_SYS_SEEK, block));
243 
244  /* At this point ptr is the current file position. */
245  if (res >= 0)
246  {
247  pfd->pos = ptr;
248  return ptr;
249  }
250  else
251  {
252  return -1;
253  }
254 }

256 static int
257 __semihosting_stat (int fd, struct stat* st)
258 {
259  struct fdent *pfd;
260  pfd = __semihosting_findslot (fd);
261  if (pfd == NULL)
262  {
263  errno = EBADF;
264  return -1;
265  }
266 
267  /* Always assume a character device,
268  with 1024 byte blocks. */
269  st->st_mode |= S_IFCHR;
270  st->st_blksize = 1024;
271 
272  int res;
273  res = __semihosting_checkerror (
274  call_host (SEMIHOSTING_SYS_FLEN, &pfd->handle));
275  if (res == -1)
276  {
277  return -1;
278  }
279 
280  /* Return the file size. */
281  st->st_size = res;
282  return 0;
283 }

298 int
299 __posix_open (const char* path, int oflag, ...)
300 {
301 
302  int fd = __semihosting_newslot ();
303  if (fd == -1)
304  {
305  errno = EMFILE;
306  return -1;
307  }
308 
309  /* It is an error to open a file that already exists. */
310  if ((oflag & O_CREAT) && (oflag & O_EXCL))
311  {
312  struct stat st;
313  int res;
314  res = __posix_stat (path, &st);
315  if (res != -1)
316  {
317  errno = EEXIST;
318  return -1;
319  }
320  }
321 
322  int aflags = 0;
323  /* The flags are Unix-style, so we need to convert them. */
324 #ifdef O_BINARY
325  if (oflag & O_BINARY)
326  {
327  aflags |= 1;
328  }
329 #endif
330 
331  /* In O_RDONLY we expect aflags == 0. */
332 
333  if (oflag & O_RDWR)
334  {
335  aflags |= 2;
336  }
337 
338  if ((oflag & O_CREAT) || (oflag & O_TRUNC) || (oflag & O_WRONLY))
339  {
340  aflags |= 4;
341  }
342 
343  if (oflag & O_APPEND)
344  {
345  /* Can't ask for w AND a; means just 'a'. */
346  aflags &= ~4;
347  aflags |= 8;
348  }
349 
350  uint32_t block[3];
351  block[0] = (uint32_t) path;
352  block[2] = std::strlen (path);
353  block[1] = (uint32_t) aflags;
354 
355  int fh = call_host (SEMIHOSTING_SYS_OPEN, block);
356 
357  /* Return a user file descriptor or an error. */
358  if (fh >= 0)
359  {
360  openfiles[fd].handle = fh;
361  openfiles[fd].pos = 0;
362  return fd;
363  }
364  else
365  {
366  return __semihosting_error (fh);
367  }
368 }

370 int
371 __posix_close (int fildes)
372 {
373  struct fdent *pfd;
374  pfd = __semihosting_findslot (fildes);
375  if (pfd == NULL)
376  {
377  errno = EBADF;
378  return -1;
379  }
380 
381  // Handle stderr == stdout.
382  if ((fildes == 1 || fildes == 2)
383  && (openfiles[1].handle == openfiles[2].handle))
384  {
385  pfd->handle = -1;
386  return 0;
387  }
388 
389  int block[1];
390  block[0] = pfd->handle;
391 
392  // Attempt to close the handle.
393  int res;
394  res = __semihosting_checkerror (call_host (SEMIHOSTING_SYS_CLOSE, block));
395 
396  // Reclaim handle?
397  if (res == 0)
398  {
399  pfd->handle = -1;
400  }
401 
402  return res;
403 }

411 ssize_t
412 __posix_read (int fildes, void* buf, size_t nbyte)
413 {
414  struct fdent *pfd;
415  pfd = __semihosting_findslot (fildes);
416  if (pfd == NULL)
417  {
418  errno = EBADF;
419  return -1;
420  }
421 
422  int block[3];
423  block[0] = pfd->handle;
424  block[1] = (int) buf;
425  block[2] = nbyte;
426 
427  int res;
428  // Returns the number of bytes *not* written.
429  res = __semihosting_checkerror (call_host (SEMIHOSTING_SYS_READ, block));
430  if (res == -1)
431  {
432  return res;
433  }
434 
435  pfd->pos += nbyte - res;
436 
437  /* res == nbyte is not an error,
438  at least if we want feof() to work. */
439  return nbyte - res;
440 }

442 ssize_t
443 __posix_write (int fildes, const void* buf, size_t nbyte)
444 {
445  struct fdent *pfd;
446  pfd = __semihosting_findslot (fildes);
447  if (pfd == NULL)
448  {
449  errno = EBADF;
450  return -1;
451  }
452 
453  int block[3];
454 
455  block[0] = pfd->handle;
456  block[1] = (int) buf;
457  block[2] = nbyte;
458 
459  // Returns the number of bytes *not* written.
460  int res;
461  res = __semihosting_checkerror (call_host (SEMIHOSTING_SYS_WRITE, block));
462  /* Clearly an error. */
463  if (res < 0)
464  {
465  return -1;
466  }
467 
468  pfd->pos += nbyte - res;
469 
470  // Did we write 0 bytes?
471  // Retrieve errno for just in case.
472  if ((nbyte - res) == 0)
473  {
474  return __semihosting_error (0);
475  }
476 
477  return (nbyte - res);
478 }

525 int
526 __posix_stat (const char* path, struct stat* buf)
527 {
528  int fd;
529  memset (buf, 0, sizeof(*buf));
530  // The best we can do is try to open the file read only.
531  // If it exists, then we can guess a few things about it.
532  if ((fd = __posix_open (path, O_RDONLY)) == -1)
533  {
534  return -1;
535  }
536  buf->st_mode |= S_IFREG | S_IREAD;
537  int res = __semihosting_stat (fd, buf);
538  // Not interested in the error.
539  __posix_close (fd);
540  return res;
541 }

572 int
573 __posix_system (const char *command)
574 {
575  // Hmmm. The ARM debug interface specification doesn't say whether
576  // SYS_SYSTEM does the right thing with a null argument, or assign any
577  // meaning to its return value. Try to do something reasonable....
578  if (command == nullptr)
579  {
580  return 1; // maybe there is a shell available? we can hope. :-P
581  }
582 
583  uint32_t block[2];
584  block[0] = (uint32_t) command;
585  block[1] = strlen (command);
586  int e = __semihosting_checkerror (call_host (SEMIHOSTING_SYS_SYSTEM, block));
587  if ((e >= 0) && (e < 256))
588  {
589  // We have to convert e, an exit status to the encoded status of
590  // the command. To avoid hard coding the exit status, we simply
591  // loop until we find the right position.
592  int exit_code;
593 
594  for (exit_code = e; (e != 0) && (WEXITSTATUS (e) != exit_code); e <<= 1)
595  {
596  continue;
597  }
598  }
599  return e;
600 }

1133 // This is the semihosting implementation for the routine to
1134 // process args.
1135 // The entire command line is received from the host
1136 // and parsed into strings.
1137 
1138 #define ARGS_BUF_ARRAY_SIZE 80
1139 #define ARGV_BUF_ARRAY_SIZE 10
1140 
1141 typedef struct command_line_block_s
1142 {
1143  char* command_line;
1144  int size;
1145 } command_line_block_t;
1146 
1147 void
1148 os_startup_initialize_args (int* p_argc, char*** p_argv)
1149 {
1150  // Array of chars to receive the command line from the host.
1151  static char args_buf[ARGS_BUF_ARRAY_SIZE];
1152 
1153  // Array of pointers to store the final argv pointers (pointing
1154  // in the above array).
1155  static char* argv_buf[ARGV_BUF_ARRAY_SIZE];
1156 
1157  int argc = 0;
1158  int is_in_argument = 0;
1159 
1160  command_line_block_t cmd_block;
1161  cmd_block.command_line = args_buf;
1162  cmd_block.size = sizeof(args_buf) - 1;
1163 
1164  int ret = call_host (SEMIHOSTING_SYS_GET_CMDLINE, &cmd_block);
1165  if (ret == 0)
1166  {
1167  // In case the host send more than we can chew, limit the
1168  // string to our buffer.
1169  args_buf[ARGS_BUF_ARRAY_SIZE - 1] = '\0';
1170 
1171  // The command line is a null terminated string.
1172  char* p = cmd_block.command_line;
1173 
1174  int delim = '\0';
1175  int ch;
1176 
1177  while ((ch = *p) != '\0')
1178  {
1179  if (is_in_argument == 0)
1180  {
1181  if (!isblank (ch))
1182  {
1183  if (argc
1184  >= (int) ((sizeof(argv_buf) / sizeof(argv_buf[0])) - 1))
1185  break;
1186 
1187  if (ch == '"' || ch == '\'')
1188  {
1189  // Remember the delimiter to search for the
1190  // corresponding terminator.
1191  delim = ch;
1192  ++p; // skip the delimiter.
1193  ch = *p;
1194  }
1195  // Remember the arg beginning address.
1196  argv_buf[argc++] = p;
1197  is_in_argument = 1;
1198  }
1199  }
1200  else if (delim != '\0')
1201  {
1202  if ((ch == delim))
1203  {
1204  delim = '\0';
1205  *p = '\0';
1206  is_in_argument = 0;
1207  }
1208  }
1209  else if (isblank (ch))
1210  {
1211  delim = '\0';
1212  *p = '\0';
1213  is_in_argument = 0;
1214  }
1215  ++p;
1216  }
1217  }
1218 
1219  if (argc == 0)
1220  {
1221  // No args found in string, return a single empty name.
1222  args_buf[0] = '\0';
1223  argv_buf[0] = &args_buf[0];
1224  ++argc;
1225  }
1226 
1227  // Must end the array with a null pointer.
1228  argv_buf[argc] = NULL;
1229 
1230  *p_argc = argc;
1231  *p_argv = &argv_buf[0];
1232 
1233  // temporary here
1234  initialise_monitor_handles ();
1235 
1236  return;
1237 }

1245 void
1246 initialise_monitor_handles (void)
1247 {
1248  // Open the standard file descriptors by opening the special
1249  // teletype device, ":tt", read-only to obtain a descriptor for
1250  // standard input and write-only to obtain a descriptor for standard
1251  // output. Finally, open ":tt" in append mode to obtain a descriptor
1252  // for standard error. Since this is a write mode, most kernels will
1253  // probably return the same value as for standard output, but the
1254  // kernel can differentiate the two using the mode flag and return a
1255  // different descriptor for standard error.
1256 
1257  int volatile block[3];
1258 
1259  block[0] = (int) ":tt";
1260  block[2] = 3; // length of filename
1261  block[1] = 0; // mode "r"
1262  monitor_stdin = call_host (SEMIHOSTING_SYS_OPEN, (void*) block);
1263 
1264  block[0] = (int) ":tt";
1265  block[2] = 3; // length of filename
1266  block[1] = 4; // mode "w"
1267  monitor_stdout = call_host (SEMIHOSTING_SYS_OPEN, (void*) block);
1268 
1269  block[0] = (int) ":tt";
1270  block[2] = 3; // length of filename
1271  block[1] = 8; // mode "a"
1272  monitor_stderr = call_host (SEMIHOSTING_SYS_OPEN, (void*) block);
1273 
1274  // If we failed to open stderr, redirect to stdout.
1275  if (monitor_stderr == -1)
1276  {
1277  monitor_stderr = monitor_stdout;
1278  }
1279 
1280  for (int i = 0; i < OS_INTEGER_SEMIHOSTING_MAX_OPEN_FILES; i++)
1281  {
1282  openfiles[i].handle = -1;
1283  }
1284

1286  openfiles[0].pos = 0;
1287  openfiles[1].handle = monitor_stdout;
1288  openfiles[1].pos = 0;
