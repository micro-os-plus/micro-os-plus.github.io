50 // This is the early hardware initialisation routine, it can be
51 // redefined in the application for more complex cases that
52 // require early inits (before BSS init).
53 //
54 // Called early from _start(), right before data & bss init.
55 //
56 // After Reset the Cortex-M processor is in Thread mode,
57 // priority is Privileged, and the Stack is set to Main.
58 
59 void __attribute__((weak))
60 os_initialize_hardware_early (void)
61 {
62  // Call the CSMSIS system initialisation routine.
63  SystemInit ();
64 
65 #if defined(__ARM_ARCH_7M__) || defined(__ARM_ARCH_7EM__)
66  // Set VTOR to the actual address, provided by the linker script.
67  // Override the manual, possibly wrong, SystemInit() setting.
68  SCB->VTOR = (uint32_t) (&__vectors_start);
69  // Ensure all subsequence instructions use the new configuration.
70  __DSB ();
71 #endif
72 
73  // The current version of SystemInit() leaves the value of the clock
74  // in a RAM variable (SystemCoreClock), which will be cleared shortly,
75  // so it needs to be recomputed after the RAM initialisations
76  // are completed.
77 
78 #if defined(OS_INCLUDE_STARTUP_INIT_FP) || (defined (__VFP_FP__) && !defined (__SOFTFP__))
79 
80  // Normally FP init is done by SystemInit(). In case this is not done
81  // there, it is possible to force its inclusion by defining
82  // OS_INCLUDE_STARTUP_INIT_FP.
83 
84  // Enable the Cortex-M4 FPU only when -mfloat-abi=hard.
85  // Code taken from Section 7.1, Cortex-M4 TRM (DDI0439C)
86 
87  // Set bits 20-23 to enable CP10 and CP11 coprocessor.
88  SCB->CPACR |= (0xF << 20);
89 
90  // Lazy save.
91  FPU->FPCCR |= FPU_FPCCR_ASPEN_Msk | FPU_FPCCR_LSPEN_Msk;
92 
93 #endif // (__VFP_FP__) && !(__SOFTFP__)
94 
95 #if defined(OS_DEBUG_SEMIHOSTING_FAULTS)
96  SCB->SHCSR |= SCB_SHCSR_USGFAULTENA_Msk;
97 #endif
98 }

100 // This is the second hardware initialisation routine, it can be
101 // redefined in the application for more complex cases that
102 // require custom inits (before constructors), otherwise these can
103 // be done in main().
104 //
105 // Called from _start(), right after data & bss init, before
106 // constructors.
107 
108 void __attribute__((weak))
109 os_initialize_hardware (void)
110 {
111  // Call the CSMSIS system clock routine to store the clock frequency
112  // in the SystemCoreClock global RAM location.
113  SystemCoreClockUpdate ();
114 }
