350  result_t
351  clock::internal_wait_until_ (timestamp_t timestamp,
352  internal::clock_timestamps_list& list)
353  {
354  thread& crt_thread = this_thread::thread ();
355 
356  // Prepare a list node pointing to the current thread.
357  // Do not worry for being on stack, it is temporarily linked to the
358  // list and guaranteed to be removed before this function returns.
359  internal::timeout_thread_node node
360  { timestamp, crt_thread };
361 
362  {
363  // ----- Enter critical section -------------------------------------
364  interrupts::critical_section ics;
365 
366  // Remove this thread from the ready list, if there.
367  port::this_thread::prepare_suspend ();
368 
369  // Add this thread to the clock waiting list.
370  list.link (node);
371  crt_thread.clock_node_ = &node;
372  crt_thread.state_ = thread::state::suspended;
373  // ----- Exit critical section --------------------------------------
374  }
375 
376  port::scheduler::reschedule ();
377 
378  {
379  // ----- Enter critical section -------------------------------------
380  interrupts::critical_section ics;
381 
382  // Remove the thread from the clock timeout list,
383  // if not already removed by the timer.
384  crt_thread.clock_node_ = nullptr;
385  node.unlink ();
386  // ----- Exit critical section --------------------------------------
387  }
388 
389  return result::ok;
390  }
